16  package extensions;
17  
18  import org.junit.jupiter.api.extension.AfterEachCallback;
19  import org.junit.jupiter.api.extension.BeforeEachCallback;
20  import org.junit.jupiter.api.extension.Extension;
21  import org.junit.jupiter.api.extension.ExtensionContext;
22  
23  import java.util.*;
24  import java.util.concurrent.*;
25  import java.util.concurrent.atomic.AtomicInteger;
26  
27  import static org.assertj.core.api.Assertions.assertThat;
28  
29  public class ExecutorExtension implements Extension, BeforeEachCallback, AfterEachCallback {
30    private TestThreadFactory threadFactory;
31    private ExecutorService executor;
32    private final List<Future<?>> futuresToPrintOnFailure = new ArrayList<>();
33  
34    public ExecutorService getExecutorService() {
35      return executor;
36    }
37  
38    @Override
39    public void beforeEach(ExtensionContext context) {
40      String displayName = context.getDisplayName();
41      threadFactory = new TestThreadFactory(displayName);
42      executor = createExecutor(threadFactory);
43    }
44  
45    @Override
46    public void afterEach(ExtensionContext context) throws Exception {
47      try {
48        executor.shutdown();
49        if (!executor.awaitTermination(5000, TimeUnit.SECONDS)) {
50          throw new Exception("ExecutorService.shutdown timed out after 5 second");
51        }
52        threadFactory.verifyAllThreadsTerminatedSuccessfully();
53        if (context.getExecutionException().isPresent()) {
54          handleFailure();
55        }
56      } catch (Throwable throwable) {
57        handleFailure();
58        throw throwable;
59      }
60    }
61  
62    private void handleFailure() {
63      threadFactory.dumpAllThreads();
64      printFuturesForFailure();
65    }
66  
67    private ExecutorService createExecutor(ThreadFactory threadFactory) {
68      return Executors.newCachedThreadPool(threadFactory);
69    }
70  
71    public void printOnFailure(List<Future<?>> futures) {
72      futuresToPrintOnFailure.addAll(futures);
73    }
74  
75    public void printOnFailure(Future<?> future) {
76      futuresToPrintOnFailure.add(future);
77    }
78  
79    private void printFuturesForFailure() {
80      System.err.println(
81          "\n===[ Dumping all registered futures ]===\n");
82      for (Future<?> future : futuresToPrintOnFailure) {
83        System.err.printf(
84            "future = %s, isDone? %s, isCancelled? %s%n",
85            future,
86            future.isDone(),
87            future.isCancelled());
88        if (future.isDone()) {
89          System.err.print("    result: ");
90          try {
91            System.err.println(future.get());
92          } catch (Exception e) {
93            e.printStackTrace();
94          }
95        }
96      }
97      System.err.println(
98          "\n===[ End dumping all registered futures ]===\n");
99    }
100 
101   private static class TestThreadFactory implements ThreadFactory {
102     private final String testName;
103     private final AtomicInteger threadCounter = new AtomicInteger();
104     private final List<Thread> threads =
105         Collections.synchronizedList(new ArrayList<>());
106 
107     private TestThreadFactory(String testName) {
108       this.testName = testName;
109     }
110 
111     @Override
112     public Thread newThread(Runnable runnable) {
113       int id = threadCounter.incrementAndGet();
114       Thread thread = new Thread(runnable, "TestThread#" + id + "[" + testName + "]");
115       threads.add(thread);
116       return thread;
117     }
118 
119     void verifyAllThreadsTerminatedSuccessfully() {
120       synchronized (threads) {
121         for (Thread thread : threads) {
122           // The Thread.State is updated asynchronously by the JVM,
123           // so we occasionally have to do a couple of retries before we
124           // observe the state change.
125           Thread.State state = thread.getState();
126           int tries = 100;
127           while (state != Thread.State.TERMINATED && tries --> 0) {
128             try {
129               thread.join(10);
130             } catch (InterruptedException e) {
131               throw new AssertionError(e);
132             }
133             state = thread.getState();
134           }
135           if (tries == 0) {
136             // Okay, this is odd. Let's ask everybody to come to a safe-point
137             // before we pass our final judgement on the thread state.
138             System.gc();
139             state = thread.getState();
140           }
141           assertThat(state).as("Unexpected thread state: " + thread + " (id " + thread.getId() + ")")
142               .isEqualTo(Thread.State.TERMINATED);
143         }
144       }
145     }
146 
147     void dumpAllThreads() {
148       synchronized (threads) {
149         System.err.println(
150             "\n===[ Dumping stack traces for all created threads ]===\n");
151         for (Thread thread : threads) {
152           StackTraceElement[] stackTrace = thread.getStackTrace();
153           printStackTrace(thread, stackTrace);
154         }
155         System.err.println(
156             "\n===[ End stack traces for all created threads ]===\n");
157 
158         System.err.println(
159             "\n===[ Dumping stack traces for all other threads ]===\n");
160         Set<Map.Entry<Thread, StackTraceElement[]>> entries =
161             Thread.getAllStackTraces().entrySet();
162         for (Map.Entry<Thread,StackTraceElement[]> entry : entries) {
163           printStackTrace(entry.getKey(), entry.getValue());
164         }
165         System.err.println(
166             "\n===[ End stack traces for all other threads ]===\n");
167       }
168     }
169 
170     private void printStackTrace(
171         Thread thread,
172         StackTraceElement[] stackTrace) {
173       Exception printer = new Exception(
174           "Stack trace for " + thread + " (id " + thread.getId() +
175               "), state = " + thread.getState());
176       printer.setStackTrace(stackTrace);
177       printer.printStackTrace();
178     }
179   }
