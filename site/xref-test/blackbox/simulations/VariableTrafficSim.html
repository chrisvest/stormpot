16  package blackbox.simulations;
17  
18  import stormpot.*;
19  
20  import java.util.concurrent.TimeUnit;
21  
22  /**
23   * A simulation that explores how the pool reacts to periods of low activity.
24   */
25  @Sim.Simulation(measurementTime = 11, output = Sim.Output.summary)
26  public class VariableTrafficSim extends Sim {
27    private static final Timeout timeout = new Timeout(1, TimeUnit.MINUTES);
28    private volatile long startTime;
29  
30    @Conf(Param.expiration)
31    public Expiration<Poolable> expiration =
32        Expiration.after(6, TimeUnit.SECONDS);
33  
34    @Conf(Param.backgroundExpirationEnabled)
35    public boolean[] backgroundExpiration = {true, false};
36  
37    // We have a number of agents...
38    @Agents({@Agent, @Agent, @Agent, @Agent})
39    public void claimRelease(Pool<Poolable> pool) throws InterruptedException {
40      if (startTime == 0) {
41        startTime = TimeUnit.NANOSECONDS.toMillis(System.nanoTime());
42      }
43      pool.claim(timeout).release();
44    }
45  
46    // ... that for an initial period are very busy ...
47    @AgentPause()
48    public long pause() {
49      long now = TimeUnit.NANOSECONDS.toMillis(System.nanoTime()) - startTime;
50      if (now < 5000) {
51        return 100; // short pause for the first 5 seconds
52      }
53  
54      // ... then suddenly they sleep for 5 seconds ...
55      if (now < 6000) {
56        // ... while they sleep, all the objects will expire at the 6 second point ...
57        return 5000;
58        // ... this leaves 3 seconds for any background allocation to replace the expired objects ...
59      }
60  
61      // ... and then they suddenly become very active again ...
62      return 100;
63    }
64  
65    // ... if no expiration checks happen in the background ...
66    @AllocationCost
67    public long allocationCost() {
68      // ... then this high allocation cost will hit the agents hard, when they wake up.
69      return 500;
70      // Each object costs 500 milliseconds to allocate, so allocating enough objects to
71      // satisfy the agents will take 2 seconds.
72    }
73  }
