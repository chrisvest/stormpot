16  package blackbox.simulations;
17  
18  import stormpot.*;
19  
20  import java.util.concurrent.TimeUnit;
21  
22  /**
23   * This simulation explores the latencies experienced when the pool is exposed
24   * to a predictable, constant throughput, with or without background
25   * expiration.
26   */
27  @Sim.Simulation(measurementTime = 60)
28  public class ConstantTrafficBackgroundExpirationSim extends Sim {
29  
30    private static final Timeout timeout = new Timeout(1, TimeUnit.SECONDS);
31  
32    @Conf(Param.expiration)
33    public Expiration<Poolable> expiration =
34        Expiration.after(200, 1000, TimeUnit.MILLISECONDS);
35  
36    @Conf(Param.backgroundExpirationEnabled)
37    public boolean[] backgroundExpiration = {true, false};
38  
39    @Agent
40    public void claimRelease(Pool<Poolable> pool) throws InterruptedException {
41      pool.claim(timeout).release();
42    }
43  
44    @AllocationCost
45    public long allocationCost() {
46      return 10;
47    }
48  
49    @AgentPause
50    public long sleep() {
51      return 100;
52    }
53  }
