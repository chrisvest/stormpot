93     * We test for this by depleting a pool, upping the size and then claiming

109   /**
110    * We must somehow ensure that the pool starts deallocating more than it
111    * allocates, when the pool is shrunk. This is difficult because the pool
112    * cannot tell us when it reaches the target size, so we have to figure this
113    * out by using a special allocator.
114    * 
115    * We test for this by configuring a CountingAllocator that also unparks a
116    * thread (namely ours, the main thread for the test) at every allocation
117    * and deallocation. We also configure the pool to have a somewhat large
118    * initial size, so we can shrink it later. Then we deplete the pool, and
119    * set a smaller target size. After setting the new target size, we release
120    * just enough objects for the pool to reach it, and then we wait for the
121    * allocator to register that same number of deallocations. This has to
122    * happen before the test times out. After that, we check that the difference
123    * between the allocations and the deallocations matches the new target size.
124    * @param fixture
125    * @throws Exception
126    */
127   @Test(timeout = 300)
128   @Theory public void
129   decreasingSizeMustEventuallyDeallocateSurplusObjects(PoolFixture fixture)
130       throws Exception {
131     int startingSize = 5;
132     int newSize = 1;
133     final Thread main = Thread.currentThread();
134     CountingAllocator allocator = new UnparkingCountingAllocator(main);
135     config.setSize(startingSize);
136     config.setAllocator(allocator);
137     ResizablePool<GenericPoolable> pool = resizable(fixture);
138     List<GenericPoolable> objs = new ArrayList<GenericPoolable>();
139     
140     while (allocator.allocations() != startingSize) {
141       objs.add(pool.claim(longTimeout)); // force the pool to do work
142     }
143     pool.setTargetSize(newSize);
144     while (allocator.deallocations() != startingSize - newSize) {
145       if (objs.size() > 0) {
146         objs.remove(0).release(); // give the pool objects to deallocate
147       } else {
148         pool.claim(longTimeout).release(); // prod it & poke it
149       }
150       LockSupport.parkNanos(10000000); // 10 millis
151     }
152     assertThat(
153         allocator.allocations() - allocator.deallocations(), is(newSize));
154   }

156   /**
157    * Similar to the decreasingSizeMustEventuallyDeallocateSurplusObjects test
158    * above, but this time the objects are all expired after the pool has been
159    * shrunk.
160    * 
161    * Again, we deplete the pool. Once depleted, our expiration has been
162    * configured such, that all subsequent items one tries to claim, will be
163    * expired.
164    * 
165    * Then we set the new lower target size, and release just enough for the
166    * pool to reach the new target.
167    * 
168    * Then we try to claim an object from the pool with a very short timeout.
169    * This will return null because the pool is still depleted. We also check
170    * that the pool has not made any new allocations, even though we have been
171    * releasing objects. We don't check the deallocations because it's
172    * complicated and we did it in the
173    * decreasingSizeMustEventuallyDeallocateSurplusObjects test above.
174    * 
175    * @param fixture
176    * @throws Exception
177    */
178   @Test(timeout = 300)
179   @Theory public void
180   mustNotReallocateWhenReleasingExpiredObjectsIntoShrunkPool(PoolFixture fixture)
181       throws Exception {
182     int startingSize = 5;
183     int newSize = 1;
184     CountingAllocator allocator = new CountingAllocator();
185     Expiration<Poolable> expiration = new CountingExpiration(
186         // our 5 items are not expired when we deplete the pool
187         false, false, false, false, false,
188         // but all items we try to claim after that *are* expired.
189         true
190         );
191     config.setExpiration(expiration).setAllocator(allocator);
192     config.setSize(startingSize);
193     ResizablePool<GenericPoolable> pool = resizable(fixture);
194     List<GenericPoolable> objs = new ArrayList<GenericPoolable>();
195     for (int i = 0; i < startingSize; i++) {
196       objs.add(pool.claim(longTimeout));
197     }
198     assertThat(objs.size(), is(startingSize));
199     pool.setTargetSize(newSize);
200     for (int i = 0; i < startingSize - newSize; i++) {
201       // release the surplus expired objects back into the pool
202       objs.remove(0).release();
203     }
204     // now the released objects should not cause reallocations, so claim
205     // returns null (it's still depleted) and allocation count stays put
206     assertThat(pool.claim(shortTimeout), nullValue());
207     assertThat(allocator.allocations(), is(startingSize));
208   }
