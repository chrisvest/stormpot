16  package stormpot;
17  
18  import java.util.concurrent.BlockingQueue;
19  import java.util.concurrent.LinkedBlockingQueue;
20  
21  import org.junit.Before;
22  import org.junit.Rule;
23  import org.junit.Test;
24  import org.junit.rules.TestRule;
25  
26  /**
27   * In this test, we make sure that the shut down process takes precautions
28   * against the possibility that it might poll a null from the dead queue.
29   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
30   */
31  public abstract class AllocThread_ShutdownNullsPool_TestTemplate<
32    SLOT,
33    ALLOC_THREAD extends Thread> {
34    @Rule public final TestRule failurePrinter = new FailurePrinterTestRule();
35    
36    protected Config<Poolable> config;
37  
38    @Before
39    public void setUp() {
40      config = new Config<Poolable>();
41      config.setAllocator(new CountingAllocator());
42      config.setSize(2);
43    }
44  
45    protected abstract ALLOC_THREAD createAllocThread(
46        BlockingQueue<SLOT> live, BlockingQueue<SLOT> dead);
47    
48    protected abstract SLOT createSlot(BlockingQueue<SLOT> live);
49  
50    @Test(timeout = 300) public void
51    mustHandleDeadNullsInShutdown() throws InterruptedException {
52      BlockingQueue<SLOT> live = createInterruptingBlockingQueue();
53      BlockingQueue<SLOT> dead = new LinkedBlockingQueue<SLOT>();
54      Thread thread = createAllocThread(live, dead);
55      thread.run();
56      // must complete before test times out, and not throw NPE
57    }
58  
59    @Test(timeout = 300) public void
60    mustHandleLiveNullsInShutdown() throws InterruptedException {
61      BlockingQueue<SLOT> live = createInterruptingBlockingQueue();
62      BlockingQueue<SLOT> dead = new LinkedBlockingQueue<SLOT>();
63      dead.add(createSlot(live));
64      Thread thread = createAllocThread(live, dead);
65      thread.run();
66      // must complete before test times out, and not throw NPE
67    }
68  
69    @SuppressWarnings("serial")
70    protected LinkedBlockingQueue<SLOT> createInterruptingBlockingQueue() {
71      return new LinkedBlockingQueue<SLOT>() {
72        public boolean offer(SLOT e) {
73          Thread.currentThread().interrupt();
74          return super.offer(e);
75        }
76      };
77    }
78  }
