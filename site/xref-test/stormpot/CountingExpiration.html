16  package stormpot;
17  
18  import java.util.concurrent.atomic.AtomicInteger;
19  
20  /**
21   * An Expiration that counts its calls and returns pre-programmed responses.
22   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
23   */
24  public class CountingExpiration implements Expiration<Poolable> {
25    private final boolean[] replies;
26    private final AtomicInteger counter;
27  
28    public CountingExpiration(boolean... replies) {
29      this.replies = replies;
30      counter = new AtomicInteger();
31    }
32  
33    @Override
34    public boolean hasExpired(SlotInfo<? extends Poolable> info) {
35      int count = counter.getAndIncrement();
36      int index = Math.max(count, replies.length - 1);
37      return replies[index];
38    }
39  
40    public int getCount() {
41      return counter.get();
42    }
43  }
