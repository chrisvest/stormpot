33  public class UnitKit {
34    private static final ExecutorService executor = Executors.newCachedThreadPool();

36    public static Thread fork(Callable<?> procedure) {
37      Thread thread = new Thread(asRunnable(procedure));
38      thread.start();
39      return thread;
40    }
41    
42    public static <T> Future<T> forkFuture(Callable<T> proceduce) {
43      return executor.submit(proceduce);
44    }

46    public static Runnable asRunnable(final Callable<?> procedure) {
47      return new Runnable() {
48        public void run() {
49          try {
50            procedure.call();
51          } catch (Exception e) {
52            throw new RuntimeException(e);
53          }
54        }
55      };
56    }
57  
58    public static <T extends Poolable> Callable<T> $claim(
59        final Pool<T> pool, final Timeout timeout) {
60      return new Callable<T>() {
61        public T call() {
62          try {
63            return pool.claim(timeout);
64          } catch (RuntimeException e) {
65            throw e;
66          } catch (InterruptedException e) {
67            throw new PoolException("claim interrupted", e);
68          }
69        }
70      };
71    }
72    
73    public static <T> Callable<T> capture(
74        final Callable<T> callable,
75        final AtomicReference<T> ref) {
76      return new Callable<T>() {
77        @Override
78        public T call() throws Exception {
79          T obj = callable.call();
80          ref.set(obj);
81          return obj;
82        }
83      };
84    }
85    
86    public static Callable<Completion> $await(
87        final Completion completion,
88        final Timeout timeout) {
89      return new Callable<Completion>() {
90        public Completion call() throws Exception {
91          completion.await(timeout);
92          return completion;
93        }
94      };
95    }

97    public static Callable<AtomicBoolean> $await(
98        final Completion completion,
99        final Timeout timeout,
100       final AtomicBoolean result) {
101     return new Callable<AtomicBoolean>() {
102       public AtomicBoolean call() throws Exception {
103         result.set(completion.await(timeout));
104         return result;
105       }
106     };
107   }

109   public static <T> Callable<T> $catchFrom(
110       final Callable<T> procedure, final AtomicReference<Exception> caught) {
111     return new Callable<T>() {
112       public T call() {
113         try {
114           return procedure.call();
115         } catch (Exception e) {
116           caught.set(e);
117         }
118         return null;
119       }
120     };
121   }

123   public static Callable<Void> $interruptUponState(
124       final Thread thread, final Thread.State state) {
125     return new Callable<Void>() {
126       public Void call() throws Exception {
127         waitForThreadState(thread, state);
128         thread.interrupt();
129         return null;
130       }
131     };
132   }
133   
134   public static Callable<Void> $delayedReleases(
135       final Poolable[] objs, long delay, TimeUnit delayUnit) {
136     final long deadline =
137       System.currentTimeMillis() + delayUnit.toMillis(delay);
138     return new Callable<Void>() {
139       public Void call() throws Exception {
140         for (Poolable obj : objs) {
141           LockSupport.parkUntil(deadline);
142           obj.release();
143         }
144         return null;
145       }
146     };
147   }

149   public static void waitForThreadState(Thread thread, Thread.State targetState) {
150     State currentState = thread.getState();
151     while (currentState != targetState) {
152       assertThat(currentState, is(not(Thread.State.TERMINATED)));
153       Thread.yield();
154       currentState = thread.getState();
155     }
156   }
157   
158   public static void join(Thread thread) {
159     try {
160       thread.join();
161     } catch (InterruptedException e) {
162       Thread.currentThread().interrupt();
163     }
164   }
165   
166   public static void join(Future<?> future) {
167     try {
168       future.get();
169     } catch (InterruptedException e) {
170       Thread.currentThread().interrupt();
171     } catch (ExecutionException e) {
172       // AssertionError doesn't have the (String, Throwable) constructor :(
173       throw new Error("Unexpected exception", e);
174     }
175   }

177   public static Completion shutdown(Pool<?> pool) {
178     assumeThat(pool, instanceOf(LifecycledPool.class));
179     return ((LifecycledPool<?>) pool).shutdown();
180   }

182   public static void spinwait(long ms) {
183     long now = System.currentTimeMillis();
184     long deadline = now + ms;
185     while (now < deadline) {
186       now = System.currentTimeMillis();
187     }
188   }
