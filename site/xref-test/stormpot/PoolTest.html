47  /**
48   * This is the generic test for Pool implementations. The test ensures that
49   * an implementation adheres to the general contract of the Pool interface,
50   * given certain circumstances and standardised configurations.
51   * <p>
52   * Pools may have other properties, and may be configurable to deviate from
53   * the standardised behaviour. However, such properties must not be
54   * observable within the standardised spectrum of configurations.
55   * <p>
56   * The tests for any special properties that a pool may have, must be put in
57   * a pool-specific test case. Do not use assumptions or other tricks to
58   * pollute this test case with tests for pool-specific or non-standard
59   * behaviours and configurations.
60   * <p>
61   * The test case uses theories to apply to the set of possible Pool
62   * implementations. Each implementation must have a PoolFixture, which is
63   * used to construct and initialise the pool, based on a {@link Config}.

66   * LifecycledPool or not. And most interesting pools are life-cycled.

68   * test a number of behaviours, but also brings about its own set of new
69   * behaviours and flows that needs to be tested for. Those tests are also
70   * included in this test case.
71   * <p>

73   * remember to update the {@link Pool} javadoc - especially the part about

422   /**
423    * SlotInfo instances must have a means of acting as a non-contended source
424    * of random numbers. We test this by getting a hold of a SlotInfo instance,
425    * and then pulling a large quantity of random numbers from it. If the
426    * numbers are random, then they will have a roughly even split between ones
427    * and zero bits.
428    */
429   @Test(timeout = 1601)
430   @Theory public void
431   slotInfoMustBeAbleToProduceRandomNumbers(PoolFixture fixture) throws Exception {
432     final AtomicReference<SlotInfo<? extends Poolable>> slotInfoRef =
433         new AtomicReference<SlotInfo<? extends Poolable>>();
434     Expiration<Poolable> expiration = new Expiration<Poolable>() {
435       public boolean hasExpired(SlotInfo<? extends Poolable> info)
436           throws Exception {
437         slotInfoRef.set(info);
438         return false;
439       }
440     };
441     config.setExpiration(expiration);
442     createPool(fixture);
443     pool.claim(longTimeout).release(); // Now we have a SlotInfo reference.
444     SlotInfo<? extends Poolable> slotInfo = slotInfoRef.get();
445     
446     // A full suite for testing the quality of the PRNG would be excessive here.
447     // We just want a back-of-the-envelope estimate that it's random.
448     int nums = 1000000;
449     int bits = 32 * nums;
450     int ones = 0;
451     for (int i = 0; i < nums; i++) {
452       ones += Integer.bitCount(slotInfo.randomInt());
453     }
454     // In the random data that we collect, we should see a roughly even split
455     // in the bits between ones and zeros.
456     // So, if we count all the one bits and double that number, we should get
457     // a number that is very close to the total number of random bits generated.
458     double diff = Math.abs(bits - ones * 2);
459     assertThat(diff, lessThan(bits * 0.005));
460   }
