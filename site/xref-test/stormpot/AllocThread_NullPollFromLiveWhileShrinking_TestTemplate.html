16  package stormpot;
17  
18  import java.util.LinkedList;
19  import java.util.Queue;
20  import java.util.concurrent.BlockingQueue;
21  
22  import org.junit.Rule;
23  import org.junit.Test;
24  import org.junit.rules.TestRule;
25  
26  /**
27   * This is a rather esoteric case. It turned out that when a queue-based Pool
28   * was resized to become smaller, and the pool was already depleted, then
29   * it was possible for the *AllocThread to pull a null from the live queue,
30   * and try to deallocate it. This is obviously not good, because it would
31   * kill the allocation thread, halting all meaningful function of the pool.
32   * 
33   * Writing a test for this, as you can probably guess, was not easy.
34   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
35   */
36  public abstract class AllocThread_NullPollFromLiveWhileShrinking_TestTemplate<
37    SLOT,
38    ALLOC_THREAD extends Thread> {
39    @Rule public final TestRule failurePrinter = new FailurePrinterTestRule();
40    
41    @SuppressWarnings("serial")
42    protected static class Stop extends RuntimeException {}
43  
44    protected BlockingQueue<SLOT> callQueue(
45        final Queue<Callable<SLOT>> calls) {
46      return new PrimedBlockingQueue<SLOT>(calls);
47    }
48  
49    protected Callable<SLOT> ret(final SLOT slot) {
50      return new Callable<SLOT>() {
51        public SLOT call() {
52          return slot;
53        }
54      };
55    }
56  
57    protected Callable<SLOT> setSizeReturn(
58        final ALLOC_THREAD th,
59        final int size,
60        final SLOT slot) {
61      return new Callable<SLOT>() {
62        public SLOT call() {
63          setTargetSize(th, size);
64          return slot;
65        }
66      };
67    }
68  
69    protected Callable<SLOT> throwStop() {
70      return new Callable<SLOT>() {
71        public SLOT call() {
72          throw new Stop();
73        }
74      };
75    }
76  
77    protected Config<Poolable> createConfig() {
78      Config<Poolable> config = new Config<Poolable>();
79      config.setAllocator(new CountingAllocator());
80      config.setSize(2);
81      return config;
82    }
83    
84    @Test(timeout = 300) public void
85    mustNotDeallocateNullWhenSizeIsReducedAndPoolIsDepleted() {
86      Queue<Callable<SLOT>> calls = new LinkedList<Callable<SLOT>>();
87      BlockingQueue<SLOT> live = callQueue(new LinkedList<Callable<SLOT>>());
88      BlockingQueue<SLOT> dead = callQueue(calls);
89      Config<Poolable> config = createConfig();
90      ALLOC_THREAD th = createAllocThread(live, dead, config);
91      
92      calls.offer(ret(createSlot(live)));
93      calls.offer(ret(createSlot(live)));
94      calls.offer(setSizeReturn(th, 1, null));
95      calls.offer(throwStop());
96      
97      try {
98        th.run();
99      } catch (Stop _) {
100       // we're happy now
101     }
102   }
103   
104   protected abstract SLOT createSlot(BlockingQueue<SLOT> live);
105 
106   protected abstract ALLOC_THREAD createAllocThread(
107       BlockingQueue<SLOT> live,
108       BlockingQueue<SLOT> dead,
109       Config<Poolable> config);
110 
111   protected abstract void setTargetSize(
112       final ALLOC_THREAD thread,
113       final int size);
114 }
