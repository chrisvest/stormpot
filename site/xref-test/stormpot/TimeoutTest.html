16  package stormpot;
17  
18  import static org.junit.Assert.*;
19  import static org.hamcrest.Matchers.*;
20  
21  import java.util.concurrent.TimeUnit;
22  
23  import org.junit.Test;
24  
25  public class TimeoutTest {
26    @Test(expected = IllegalArgumentException.class) public void
27    timeUnitCannotBeNull() {
28      new Timeout(1, null);
29    }
30    
31    @Test public void
32    timeoutCanBeZeroOrLess() {
33      // because it means we don't want to do any waiting at all.
34      new Timeout(0, TimeUnit.DAYS);
35      new Timeout(-1, TimeUnit.DAYS);
36    }
37    
38    @Test public void
39    baseTimeUnitMustNotBeNull() {
40      assertNotNull("unexpectedly got null for the base unit",
41          new Timeout(1, TimeUnit.DAYS).getBaseUnit());
42    }
43    
44    @Test public void
45    timeoutsWithEqualValueButDifferentUnitsAreEqual() {
46      Timeout a = new Timeout(1, TimeUnit.SECONDS);
47      Timeout b = new Timeout(1000, TimeUnit.MILLISECONDS);
48      assertThat(a, equalTo(b));
49    }
50    
51    @Test public void
52    differentTimeoutValuesAreNotEqual() {
53      Timeout a = new Timeout(1, TimeUnit.SECONDS);
54      Timeout b =  new Timeout(1, TimeUnit.MILLISECONDS);
55      assertThat(a, not(equalTo(b)));
56    }
57    
58    @Test public void
59    timeoutsAreNotEqualToObjectsOfDifferentClasses() {
60      Object a = new Timeout(1, TimeUnit.SECONDS);
61      Object b = "poke";
62      assertThat(a, not(equalTo(b)));
63    }
64    
65    @Test public void
66    timeoutsAreNotEqualToNull() {
67      Timeout a = new Timeout(1, TimeUnit.SECONDS);
68      assertFalse(a.equals(null));
69    }
70    
71    @Test public void
72    timeoutsAreEqualToThemselves() {
73      Timeout a = new Timeout(1, TimeUnit.SECONDS);
74      assertThat(a, equalTo(a));
75    }
76    
77    @Test public void
78    sameValueAndUnitGivesSameHashCode() {
79      Timeout a = new Timeout(1, TimeUnit.SECONDS);
80      Timeout b = new Timeout(1, TimeUnit.SECONDS);
81      assertTrue(a.hashCode() == b.hashCode());
82    }
83    
84    @Test public void
85    differentValueGivesDifferentHashcode() {
86      Timeout a = new Timeout(1, TimeUnit.SECONDS);
87      Timeout b = new Timeout(2, TimeUnit.SECONDS);
88      assertTrue(a.hashCode() != b.hashCode());
89    }
90    
91    @Test public void
92    differentUnitsForSameValuesGivesSameHashcode() {
93      Timeout a = new Timeout(1, TimeUnit.SECONDS);
94      Timeout b = new Timeout(1000, TimeUnit.MILLISECONDS);
95      assertTrue(a.hashCode() == b.hashCode());
96    }
97    
98    @Test public void
99    sameValuesOfDifferentUnitsGivesDifferentHashcode() {
100     Timeout a = new Timeout(1, TimeUnit.SECONDS);
101     Timeout b = new Timeout(1, TimeUnit.MILLISECONDS);
102     assertTrue(a.hashCode() != b.hashCode());
103   }
104 }
