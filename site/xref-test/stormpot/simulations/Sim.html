123   public static void main(String[] args) throws Exception {
124     long start = System.currentTimeMillis();
125     String cmdClass = System.getProperty("sun.java.command").replaceFirst("^.*\\s+", "");
126     Class<?> klass = Class.forName(cmdClass);
127     Sim sim = (Sim) klass.newInstance();
128 
129     // Find all the pool implementation constructors.
130     List<Constructor<LifecycledPool<GenericPoolable>>> ctors =
131         new ArrayList<Constructor<LifecycledPool<GenericPoolable>>>();
132     Simulation simulation = klass.getAnnotation(Simulation.class);
133     if (simulation != null) {
134       for (Class<? extends LifecycledPool> type : simulation.pools()) {
135         Constructor<?> constructor = type.getConstructor(Config.class);
136         ctors.add((Constructor<LifecycledPool<GenericPoolable>>) constructor);
137       }
138     } else {
139       Constructor<?> constructor = QueuePool.class.getConstructor(Config.class);
140       ctors.add((Constructor<LifecycledPool<GenericPoolable>>) constructor);
141       constructor = BlazePool.class.getConstructor(Config.class);
142       ctors.add((Constructor<LifecycledPool<GenericPoolable>>) constructor);
143     }
144 
145     // Go through all fields, constructing a set of configurations.
146     // Run each configuration for each pool implementation.
147 
148     AtomicBoolean enableAllocationCost = new AtomicBoolean();
149     Collection<Config<GenericPoolable>> configurations =
150         buildConfigurations(klass, sim, enableAllocationCost);
151     for (Config<GenericPoolable> config : configurations) {
152       for (Constructor<LifecycledPool<GenericPoolable>> ctor : ctors) {
153         Output output = simulation == null? Output.detailed : simulation.output();
154         simulate(sim, ctor, config, enableAllocationCost, output);
155       }
156     }
157     long elapsedMillis = System.currentTimeMillis() - start;
158     System.out.printf("Done, %.3f seconds.%n", elapsedMillis / 1000.0);
159     System.exit(0);
160   }

162   private static Collection<Config<GenericPoolable>> buildConfigurations(
163       Class<?> klass, Sim sim, AtomicBoolean enableAllocationCost) throws Exception {
164     // Find all fields annotated with @Conf
165     // Group them by their Param category
166     // Produce a Config for every permutation of each group
167 
168     // Building the groups
169     EnumMap<Param, List<Object>> groups = new EnumMap<Param, List<Object>>(Param.class);
170     for (Field field : klass.getDeclaredFields()) {
171       Conf conf = field.getAnnotation(Conf.class);
172       if (conf != null) {
173         Param param = conf.value();
174         Object value = field.get(sim);
175         List<Object> values = groups.get(param);
176         if (values == null) {
177           values = new LinkedList<Object>();
178           groups.put(param, values);
179         }
180         values.add(value);
181       }
182     }
183 
184     // Sequence the groups
185     Link<Map.Entry<Param, List<Object>>> head = null;
186     for (Map.Entry<Param, List<Object>> entry : groups.entrySet()) {
187       head = new Link<Map.Entry<Param, List<Object>>>(head, entry);
188     }
189 
190     // Building Config permutations
191     Config<GenericPoolable> baseConfig = new Config<GenericPoolable>();
192     CountingAllocator allocator = buildAllocator(sim, enableAllocationCost);
193     baseConfig.setAllocator(allocator);
194     List<Config<GenericPoolable>> result = new LinkedList<Config<GenericPoolable>>();
195     addConfigPermutations(head, baseConfig, result);
196     return result;
197   }

199   private static CountingAllocator buildAllocator(
200       final Sim sim,
201       final AtomicBoolean enableAllocationCost) {
202     Class<?> klass = sim.getClass();
203     TimeUnit allocationTimeUnit = null;
204     Method allocationTimeMethod = null;
205     for (Method method : klass.getDeclaredMethods()) {
206       AllocationCost allocationCost = method.getAnnotation(AllocationCost.class);
207       if (allocationCost != null) {
208         assert allocationTimeUnit == null: "Found more than one @AllocationCost method";
209         allocationTimeUnit = allocationCost.value();
210         allocationTimeMethod = method;
211       }
212     }
213 
214     if (allocationTimeMethod == null) {
215       return allocator();
216     }
217 
218     final TimeUnit unit = allocationTimeUnit;
219     final Method method = allocationTimeMethod;
220     return allocator(alloc(new Action() {
221       @Override
222       public GenericPoolable apply(Slot slot, GenericPoolable obj) throws Exception {
223         if (enableAllocationCost.get()) {
224           Number time = (Number) method.invoke(sim);
225           unit.sleep(time.longValue());
226         }
227         return $new.apply(slot, obj);
228       }
229     }));
230   }

232   private static void addConfigPermutations(
233       Link<Map.Entry<Param, List<Object>>> head,
234       Config<GenericPoolable> baseConfig,
235       List<Config<GenericPoolable>> result) throws Exception {
236     if (head == null) {
237       // Base case
238       result.add(baseConfig);
239     } else {
240       // Reduction
241       Param param = head.value.getKey();
242       List<Object> values = head.value.getValue();
243 
244       for (Object value : values) {
245         if (value.getClass().isArray()) {
246           int length = Array.getLength(value);
247           for (int i = 0; i < length; i++) {
248             Config<GenericPoolable> mutation = baseConfig.clone();
249             param.set(mutation, Array.get(value, i));
250             addConfigPermutations(head.tail, mutation, result);
251           }
252         } else {
253           Config<GenericPoolable> mutation = baseConfig.clone();
254           param.set(mutation, value);
255           addConfigPermutations(head.tail, mutation, result);
256         }
257       }
258     }

261   private static void simulate(
262       Sim sim,
263       Constructor<LifecycledPool<GenericPoolable>> ctor,
264       Config<GenericPoolable> config,
265       AtomicBoolean enableAllocationCost,
266       Output output) throws Exception {
267     DependencyResolver deps = new DependencyResolver();
268     deps.add(config);
269     deps.add(config.getAllocator());
270     deps.add(config.getExpiration());
271     deps.add(config.getMetricsRecorder());
272     deps.add(config.getThreadFactory());
273     deps.add(output);
274 
275     String poolTypeName = ctor.getDeclaringClass().getSimpleName();
276     System.out.printf("Simulating %s %s for %s%n",
277         sim.getClass().getSimpleName(), describe(config), poolTypeName);
278 
279     long measurementTimeMillis = getMeasurementTimeMillis(sim);
280     int iterations = 100;
281     long warmupTimeMillis = measurementTimeMillis / iterations;
282     for (int i = 1; i <= iterations; i++) {
283       boolean isWarmup = i < iterations;
284       long runTime = isWarmup? warmupTimeMillis : measurementTimeMillis;
285 
286       // Reset internal state.
287       enableAllocationCost.set(false);
288       LifecycledPool<GenericPoolable> pool = ctor.newInstance(config);
289       sim = sim.getClass().newInstance();
290       deps.replace(pool);
291 
292       // Wait for the pool to boot up.
293       int size = config.getSize();
294       List<GenericPoolable> objs = new ArrayList<GenericPoolable>();
295       for (int j = 0; j < size; j++) {
296         objs.add(pool.claim(new Timeout(30, TimeUnit.MINUTES)));
297       }
298       for (GenericPoolable obj : objs) {
299         obj.release();
300       }
301 
302       // Then run.
303       enableAllocationCost.set(true);
304       sim.run(deps, runTime, isWarmup? Output.none : output);
305 
306       // Finally clean up a bit.
307       if (!pool.shutdown().await(SHUTDOWN_TIMEOUT)) {
308         ManagedPool managedPool = (ManagedPool) pool;
309         System.err.printf(
310             "Shutdown timed out! Pool type = %s, config = %s, " +
311                 "object leaks detected = %s%n",
312             poolTypeName, describe(config), managedPool.getLeakedObjectsCount());
313       }
314     }
315   }

317   private static long getMeasurementTimeMillis(Sim sim) {
318     Class<?> klass = sim.getClass();
319     Simulation simulation = klass.getAnnotation(Simulation.class);
320     if (simulation != null) {
321       long time = simulation.measurementTime();
322       TimeUnit unit = simulation.measurementTimeUnit();
323       return unit.toMillis(time);
324     }
325     return 10000;

346   private void run(
347       DependencyResolver deps,
348       long runTimeMillis,
349       Output output) throws InterruptedException {
350     // Find all agents and their corresponding configuration fields, and create
351     // a thread for each.
352     ControlSignal controlSignal = new ControlSignal();
353 
354     // Agents are defined by the @Agent or @Agents annotations.
355     // Their timings are defined by the @AgentPause annotation, linked by their name.
356     Method[] methods = getClass().getMethods();
357     List<AgentRunner> agents = new ArrayList<AgentRunner>();
358     Map<String, IterationPause> iterationPauseIndex = new HashMap<String, IterationPause>();
359     for (Method method : methods) {
360       AgentPause agentPause = method.getAnnotation(AgentPause.class);
361       if (agentPause != null) {
362         IterationPause pause = new IterationPause(agentPause, this, method);
363         iterationPauseIndex.put(agentPause.name(), pause);
364       }
365     }
366 
367     for (Method method : methods) {
368       Agent agent = method.getAnnotation(Agent.class);
369       if (agent != null) {
370         startAgent(deps, controlSignal, agents, iterationPauseIndex, method, agent);
371       }
372       Agents agentsAnnotation = method.getAnnotation(Agents.class);
373       if (agentsAnnotation != null) {
374         for (Agent subAgent : agentsAnnotation.value()) {
375           startAgent(deps, controlSignal, agents, iterationPauseIndex, method, subAgent);
376         }
377       }
378     }
379 
380     if (agents.isEmpty()) {
381       System.err.println("No public @Agent or @Agents annotated methods found!");
382       return;
383     }
384 
385     // Then start them all, and run them for as long as need be.
386     controlSignal.start();
387     Thread.sleep(runTimeMillis);
388     controlSignal.stop();
389     controlSignal.awaitDone(agents.size());
390 
391     // Finally print their results, if need be.
392     if (output != Output.none) {
393       Histogram sum = newHistogram();
394       for (AgentRunner agent : agents) {
395         sum.add(agent.histogram);
396         if (output == Output.detailed) {
397           agent.printResults();
398         }
399       }
400       if (agents.size() > 1 || output == Output.summary) {
401         System.out.println("Latency results sum:");
402         printHistogram(sum);
403         System.out.println();
404       }
405     }
406   }

408   private void startAgent(
409       DependencyResolver deps,
410       ControlSignal controlSignal,
411       List<AgentRunner> agents,
412       Map<String, IterationPause> iterationPauseIndex,
413       Method method,
414       Agent agent) {
415     IterationPause iterationPause = iterationPauseIndex.get(agent.name());
416     AgentRunner runner = new AgentRunner(
417         controlSignal, deps, agent, this, method, iterationPause);
418     Thread thread = new Thread(runner, "Agent[" + agent.name() + "]");
419     thread.start();
420     agents.add(runner);
421   }

423   private static class AgentRunner implements Runnable {
424     private final ControlSignal controlSignal;
425     private final long initialDelayTime;
426     private final TimeUnit initialDelayUnit;
427     private final String agentName;
428     private final Sim sim;
429     private final Method agentMethod;
430     private final IterationPause iterationPause;
431     private final Histogram histogram;
432     private final Object[] argumentList;
433 
434     public AgentRunner(
435         ControlSignal controlSignal,
436         DependencyResolver deps,
437         Agent agent,
438         Sim sim,
439         Method agentMethod,
440         IterationPause iterationPause) {
441       this.controlSignal = controlSignal;
442       initialDelayTime = agent.initialDelay();
443       initialDelayUnit = agent.initialDelayUnit();
444       agentName = agent.name();
445       this.sim = sim;
446       this.agentMethod = agentMethod;
447       this.iterationPause = iterationPause;
448       histogram = newHistogram();
449       argumentList = deps.resolve(agentMethod.getParameterTypes());
450     }
451 
452     @Override
453     public void run() {
454       try {
455         runUnsafe();
456       } catch (Exception e) {
457         throw new RuntimeException("Agent[" + agentName + "] failed", e);
458       }
459     }
460 
461     private void runUnsafe() throws Exception {
462       controlSignal.awaitStart();
463       if (initialDelayUnit != null) {
464         initialDelayUnit.sleep(initialDelayTime);
465       }
466 
467       long lastPauseTime = 0;
468       while (!controlSignal.stopped()) {
469         long start = System.nanoTime();
470         callAgentMethod();
471         long elapsed = System.nanoTime() - start;
472         long micros = elapsed / 1000;
473         histogram.recordValueWithExpectedInterval(micros, lastPauseTime);
474         lastPauseTime = iterationPause.pause(micros);
475       }
476       controlSignal.done();
477     }
478 
479     private void callAgentMethod() {
480       try {
481         agentMethod.invoke(sim, argumentList);
482       } catch (Exception e) {
483         throw new RuntimeException("Agent[" + agentName + "] method invocation failure", e);
484       }
485     }
486 
487     public void printResults() {
488       System.out.printf("Latency results for Agent[%s]:%n", agentName);
489       printHistogram(histogram);
490       System.out.println();
491     }
492   }

502   private static class IterationPause {
503     private final Sim sim;
504     private final Method pauseMethod;
505     private final TimeUnit pauseUnit;
506 
507     public IterationPause(AgentPause agentPause, Sim sim, Method pauseMethod) {
508       this.sim = sim;
509       this.pauseMethod = pauseMethod;
510       this.pauseUnit = agentPause.unit();
511     }
512 
513     public long pause(long spentTimeMicros) throws Exception {
514       Number result = (Number) pauseMethod.invoke(sim);
515       long pauseTime = result.longValue();
516       long upauseTime = pauseUnit.toMicros(pauseTime);
517       TimeUnit.MICROSECONDS.sleep(upauseTime - spentTimeMicros);
518       return upauseTime;
519     }
520   }

522   private static class ControlSignal {
523     private final CountDownLatch startLatch;
524     private final Semaphore doneCounter;
525     private volatile boolean stopped;
526 
527     private ControlSignal() {
528       startLatch = new CountDownLatch(1);
529       doneCounter = new Semaphore(0);
530     }
531 
532     public void awaitStart() throws InterruptedException {
533       startLatch.await();
534     }
535 
536     public void start() {
537       startLatch.countDown();
538     }
539 
540     public boolean stopped() {
541       return stopped;
542     }
543 
544     public void stop() {
545       stopped = true;
546     }
547 
548     public void done() {
549       doneCounter.release();
550     }
551 
552     public void awaitDone(int doneCounts) throws InterruptedException {
553       doneCounter.acquire(doneCounts);
554     }

557   private static class DependencyResolver {
558     private final List<Object> dependencies;
559 
560     private DependencyResolver() {
561       dependencies = new ArrayList<Object>();
562     }
563 
564     public void add(Object obj) {
565       if (obj != null) {
566         dependencies.add(obj);
567       }
568     }
569 
570     public Object[] resolve(Class<?>[] types) {
571       Object[] values = new Object[types.length];
572       for (int i = 0; i < values.length; i++) {
573         values[i] = resolve(types[i]);
574       }
575       return values;
576     }
577 
578     private <T> T resolve(Class<T> type) {
579       for (Object dependency : dependencies) {
580         if (type.isInstance(dependency)) {
581           return type.cast(dependency);
582         }
583       }
584       return null;
585     }
586 
587     public void replace(Object obj) {
588       int size = dependencies.size();
589       for (int i = 0; i < size; i++) {
590         if (dependencies.get(i).getClass() == obj.getClass()) {
591           dependencies.set(i, obj);
592           return;
593         }
594       }
595       add(obj);
596     }
