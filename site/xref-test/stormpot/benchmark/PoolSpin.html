16  package stormpot.benchmark;

18  import java.util.ArrayList;
19  import java.util.Collections;
20  import java.util.List;
21  import java.util.concurrent.TimeUnit;
22  import java.util.concurrent.locks.LockSupport;
23  
24  import stormpot.Allocator;
25  import stormpot.Config;
26  import stormpot.GenericPoolable;
27  import stormpot.LifecycledPool;
28  import stormpot.Pool;
29  import stormpot.PoolFixtures;
30  import stormpot.Poolable;
31  import stormpot.Slot;
32  
33  import com.google.caliper.Param;
34  import com.google.caliper.SimpleBenchmark;
35  
36  @SuppressWarnings("unchecked")
37  public class PoolSpin extends SimpleBenchmark {
38    private final class SlowAllocator implements Allocator<GenericPoolable> {
39      private final List allocated =
40        Collections.synchronizedList(new ArrayList());
41      private final List deallocated =
42        Collections.synchronizedList(new ArrayList());
43      private final long workMs;
44  
45      public SlowAllocator(long workMs) {
46        this.workMs = workMs;
47      }
48      
49      public GenericPoolable allocate(Slot slot) {
50        long deadline = System.currentTimeMillis() + workMs;
51        LockSupport.parkUntil(deadline);
52        GenericPoolable obj = new GenericPoolable(slot);
53        allocated.add(obj);
54        return obj;
55      }
56  
57      public void deallocate(GenericPoolable poolable) {
58        poolable.deallocated = true;
59        deallocated.add(poolable);
60      }
61    }
62  
63    @Param({"10"}) int size = 10;
64    @Param({"10"}) long work = 10;
65    @Param({"0", "1", "2"}) int poolType;
66    @Param({"10000"}) long ttl = 10000;
67  
68    protected Pool<GenericPoolable> pool;
69    
70    @Override
71    protected void setUp() throws Exception {
72      Config config = new Config();
73      config.setAllocator(new SlowAllocator(work));
74      config.setSize(size);
75      config.setTTL(ttl, TimeUnit.MILLISECONDS);
76      pool = PoolFixtures.poolFixtures()[poolType].initPool(config);
77      // Give the pool 500 ms to boot up any threads it might need
78      LockSupport.parkNanos(500000000);
79    }
80    
81    public int timeClaimReleaseSpin(int reps) throws Exception {
82      int result = 1235789;
83      for (int i = 0; i < reps; i++) {
84        Poolable obj = claim(pool);
85        result ^= obj.hashCode();
86        obj.release();
87      }
88      return result;
89    }
90  
91    static GenericPoolable claim(Pool<GenericPoolable> pool)
92    throws InterruptedException {
93      GenericPoolable obj = pool.claim();
94      obj.lastClaimBy = Thread.currentThread();
95      return obj;
96    }
97  
98    @Override
99    protected void tearDown() throws Exception {
100     if (pool instanceof LifecycledPool) {
101       LifecycledPool p = (LifecycledPool) pool;
102       p.shutdown().await();
103     }
104   }
