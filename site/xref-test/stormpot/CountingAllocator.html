16  package stormpot;
17  
18  import java.util.ArrayList;
19  import java.util.Collections;
20  import java.util.List;
21  import java.util.concurrent.atomic.AtomicInteger;
22  
23  public class CountingAllocator implements Allocator {
24    private final AtomicInteger allocations = new AtomicInteger();
25    private final AtomicInteger deallocations = new AtomicInteger();
26    private final List<Poolable> allocated =
27      Collections.synchronizedList(new ArrayList<Poolable>());
28    private final List<Poolable> deallocated =
29      Collections.synchronizedList(new ArrayList<Poolable>());
30  
31    public Poolable allocate(Slot slot) throws Exception {
32      allocations.incrementAndGet();
33      GenericPoolable obj = new GenericPoolable(slot);
34      allocated.add(obj);
35      return obj;
36    }
37  
38    public void deallocate(Poolable poolable) throws Exception {
39      deallocations.incrementAndGet();
40      deallocated.add(poolable);
41    }
42  
43    public void reset() {
44      allocations.set(0);
45      deallocations.set(0);
46    }
47    
48    public int allocations() {
49      return allocations.get();
50    }
51    
52    public int deallocations() {
53      return deallocations.get();
54    }
55  
56    public List<Poolable> deallocationList() {
57      return deallocated;
58    }
59  }
