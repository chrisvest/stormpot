16  package stormpot.qpool;
17  
18  import java.util.LinkedList;
19  import java.util.Queue;
20  import java.util.concurrent.BlockingQueue;
21  
22  import org.junit.Test;
23  
24  import stormpot.Config;
25  import stormpot.CountingAllocator;
26  import stormpot.Poolable;
27  
28  /**
29   * This is a rather esoteric case. It turned out that when the QueuePool
30   * was resized to become smaller, and the pool was already depleted, then
31   * it was possible for the QAllocThread to pull a null from the live queue,
32   * and try to deallocate it. This is obviously not good, because it would
33   * kill the allocation thread, halting all meaningful function of the pool.
34   * 
35   * Writing a test for this, as you can probably guess, was not easy.
36   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
37   */
38  public class QAllocThread_NullPollFromLiveWhileShrinking_Test {
39    interface Callable<T> {
40      T call();
41    }
42    
43    @SuppressWarnings("serial")
44    static class Stop extends RuntimeException {}
45  
46    private BlockingQueue<QSlot<Poolable>> callQueue(
47        final Queue<Callable<QSlot<Poolable>>> calls) {
48      return new PrimedBlockingQueue(calls);
49    }
50  
51    private Callable<QSlot<Poolable>> ret(final QSlot<Poolable> slot) {
52      return new Callable<QSlot<Poolable>>() {
53        public QSlot<Poolable> call() {
54          return slot;
55        }
56      };
57    }
58  
59    private Callable<QSlot<Poolable>> setSizeReturn(
60        final QAllocThread<Poolable> th,
61        final int size,
62        final QSlot<Poolable> slot) {
63      return new Callable<QSlot<Poolable>>() {
64        public QSlot<Poolable> call() {
65          th.setTargetSize(size);
66          return slot;
67        }
68      };
69    }
70  
71    private Callable<QSlot<Poolable>> throwStop() {
72      return new Callable<QSlot<Poolable>>() {
73        public QSlot<Poolable> call() {
74          throw new Stop();
75        }
76      };
77    }
78    
79    @Test(timeout = 300) public void
80    mustNotDeallocateNullWhenSizeIsReducedAndPoolIsDepleted() {
81      Queue<Callable<QSlot<Poolable>>> calls =
82          new LinkedList<Callable<QSlot<Poolable>>>();
83      @SuppressWarnings({ "unchecked", "rawtypes" })
84      BlockingQueue<QSlot<Poolable>> live = callQueue(new LinkedList());
85      BlockingQueue<QSlot<Poolable>> dead = callQueue(calls);
86      Config<Poolable> config = new Config<Poolable>();
87      config.setAllocator(new CountingAllocator());
88      config.setSize(2);
89      QAllocThread<Poolable> th = new QAllocThread<Poolable>(live, dead, config);
90      
91      calls.offer(ret(new QSlot<Poolable>(live)));
92      calls.offer(ret(new QSlot<Poolable>(live)));
93      calls.offer(setSizeReturn(th, 1, null));
94      calls.offer(throwStop());
95      
96      try {
97        th.run();
98      } catch (Stop _) {
99        // we're happy now
100     }
101   }
102 }
