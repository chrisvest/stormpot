16  package stormpot;
17  
18  import static org.junit.Assert.*;
19  
20  import java.util.concurrent.TimeUnit;
21  
22  import org.junit.Test;
23  
24  public class TimeExpirationTest {
25  
26    private Expiration<Poolable> createExpiration(int ttl) {
27      return new TimeExpiration(ttl, TimeUnit.MILLISECONDS);
28    }
29    
30    private SlotInfo<?> infoWithAge(final long ageMillis) {
31      return new SlotInfo<Poolable>() {
32        public long getAgeMillis() {
33          return ageMillis;
34        }
35  
36        @Override
37        public long getClaimCount() {
38          return 0;
39        }
40  
41        @Override
42        public Poolable getPoolable() {
43          return null;
44        }
45      };
46    }
47    
48    @Test(expected = IllegalArgumentException.class) public void
49    timeUnitCannotBeNull() {
50      new TimeExpiration(10, null);
51    }
52    
53    @Test public void
54    youngSlotsAreNotInvalid() {
55      Expiration<Poolable> expiration = createExpiration(2);
56      SlotInfo<?> info = infoWithAge(1);
57      assertFalse(expiration.hasExpired(info));
58    }
59  
60    @Test public void
61    slotsAtTheMaximumPermittedAgeAreNotInvalid() {
62      Expiration<Poolable> expiration = createExpiration(2);
63      SlotInfo<?> info = infoWithAge(2);
64      assertFalse(expiration.hasExpired(info));
65    }
66    
67    @Test public void
68    slotsOlderThanTheMaximumPermittedAgeAreInvalid() {
69      Expiration<Poolable> expiration = createExpiration(2);
70      SlotInfo<?> info = infoWithAge(3);
71      assertTrue(expiration.hasExpired(info));
72    }
73    
74    @Test(expected = IllegalArgumentException.class) public void
75    maxPermittedAgeCannotBeLessThanOne() {
76      createExpiration(0);
77    }
78  }
