16  package stormpot.qpool;
17  
18  import java.util.concurrent.BlockingQueue;
19  import java.util.concurrent.atomic.AtomicBoolean;
20  
21  import stormpot.Poolable;
22  import stormpot.Slot;
23  
24  class QSlot<T extends Poolable> implements Slot {
25    final BlockingQueue<QSlot<T>> live;
26    final AtomicBoolean claimed;
27    T obj;
28    Exception poison;
29    long expires;
30    
31    public QSlot(BlockingQueue<QSlot<T>> live) {
32      this.live = live;
33      this.claimed = new AtomicBoolean();
34    }
35    
36    public void claim() {
37      claimed.set(true);
38    }
39  
40    public boolean expired() {
41      return expires < System.currentTimeMillis();
42    }
43    
44    public void release(Poolable obj) {
45      if (claimed.compareAndSet(true, false)) {
46        live.offer(this);
47      }
48    }
49  }
