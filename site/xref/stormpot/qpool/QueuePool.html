16  package stormpot.qpool;
17  
18  import java.util.concurrent.BlockingQueue;
19  import java.util.concurrent.LinkedBlockingQueue;
20  
21  import stormpot.Completion;
22  import stormpot.Config;
23  import stormpot.Expiration;
24  import stormpot.LifecycledResizablePool;
25  import stormpot.PoolException;
26  import stormpot.Poolable;
27  import stormpot.Timeout;
28  
29  /**
30   * QueuePool is a fairly simple {@link LifecycledResizablePool} implementation
31   * that basically consists of a queue of Poolable instances, and a Thread to
32   * allocate them.
33   * <p>
34   * This means that the object allocation always happens in a dedicated thread.
35   * This means that no thread that calls any of the claim methods, will incur
36   * the overhead of allocating Poolables. This should lead to reduced deviation
37   * in the times it takes claim method to complete, provided the pool is not
38   * depleted.
39   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
40   * @param <T> The type of {@link Poolable} managed by this pool.
41   */
42  public final class QueuePool<T extends Poolable>
43  implements LifecycledResizablePool<T> {
44    private final BlockingQueue<QSlot<T>> live;
45    private final BlockingQueue<QSlot<T>> dead;
46    private final QAllocThread<T> allocThread;
47    private final Expiration<? super T> deallocRule;
48    private volatile boolean shutdown = false;
49    
50    /**
51     * Construct a new QueuePool instance based on the given {@link Config}.
52     * @param config The pool configuration to use.
53     */
54    public QueuePool(Config<T> config) {
55      live = new LinkedBlockingQueue<QSlot<T>>();
56      dead = new LinkedBlockingQueue<QSlot<T>>();
57      synchronized (config) {
58        config.validate();
59        allocThread = new QAllocThread<T>(live, dead, config);
60        deallocRule = config.getExpiration();
61      }
62      allocThread.start();
63    }
64  
65    private void checkForPoison(QSlot<T> slot) {
66      if (slot == allocThread.POISON_PILL) {
67        live.offer(allocThread.POISON_PILL);
68        throw new IllegalStateException("pool is shut down");
69      }
70      if (slot.poison != null) {
71        Exception poison = slot.poison;
72        dead.offer(slot);
73        throw new PoolException("allocation failed", poison);
74      }
75      if (shutdown) {
76        dead.offer(slot);
77        throw new IllegalStateException("pool is shut down");
78      }
79    }
80  
81    private boolean isInvalid(QSlot<T> slot) {
82      boolean invalid = true;
83      RuntimeException exception = null;
84      try {
85        invalid = deallocRule.hasExpired(slot);
86      } catch (RuntimeException ex) {
87        exception = ex;
88      }
89      if (invalid) {
90        // it's invalid - into the dead queue with it and continue looping
91        dead.offer(slot);
92        if (exception != null) {
93          throw exception;
94        }
95      } else {
96        // it's valid - claim it and stop looping
97        slot.claim();
98      }
99      return invalid;
100   }
101 
102   public T claim(Timeout timeout) throws PoolException,
103       InterruptedException {
104     if (timeout == null) {
105       throw new IllegalArgumentException("timeout cannot be null");
106     }
107     QSlot<T> slot;
108     long deadline = timeout.getDeadline();
109     do {
110       long timeoutLeft = timeout.getTimeLeft(deadline);
111       slot = live.poll(timeoutLeft, timeout.getBaseUnit());
112       if (slot == null) {
113         // we timed out while taking from the queue - just return null
114         return null;
115       }
116       checkForPoison(slot);
117     } while (isInvalid(slot));
118     return slot.obj;
119   }
120 
121   public Completion shutdown() {
122     shutdown = true;
123     allocThread.interrupt();
124     return new QPoolShutdownCompletion(allocThread);
125   }
126 
127   public void setTargetSize(int size) {
128     if (size < 1) {
129       throw new IllegalArgumentException("target size must be at least 1");
130     }
131     allocThread.setTargetSize(size);
132   }
133 
134   public int getTargetSize() {
135     return allocThread.getTargetSize();
136   }
137 }
