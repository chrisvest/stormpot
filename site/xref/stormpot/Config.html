16  package stormpot;

18  import java.util.concurrent.ThreadFactory;
19  import java.util.concurrent.TimeUnit;

21  /**
22   * = The basic configuration "bean" class.
23   *
24   * Instances of this class is passed to the constructors of {@link Pool pools}
25   * so that they know how big they should be, how to allocate objects and when
26   * to deallocate objects.
27   *
28   * This class is made thread-safe by having the fields be protected by the
29   * intrinsic object lock on the Config object itself. This way, pools can
30   * `synchronize` on the config object to read the values out
31   * atomically.
32   *
33   * The various set* methods are made to return the Config instance itself, so
34   * that the method calls may be chained if so desired.
35   * 
36   * = Standardised configuration
37   *
38   * The contract of the Config class, and how Pools will interpret it, is within
39   * the context of a so-called standardised configuration. All pool and Config
40   * implementations must behave similarly in a standardised configuration.
41   *
42   * It is conceivable that some pool implementations will come with their own
43   * sub-classes of Config, that allow greater control over the pools behaviour.
44   * It is even permissible that these pool implementations may
45   * deviate from the contract of the Pool interface. However, they are only
46   * allowed to do so in a non-standard configuration. That is, any deviation
47   * from the specified contracts must be explicitly enabled.
48   * 
49   * @author Chris Vest <mr.chrisvest@gmail.com>
50   * @param <T> The type of {@link Poolable} objects that a {@link Pool} based
51   * on this Config will produce.
52   */
53  @SuppressWarnings("unchecked")
54  public class Config<T extends Poolable> implements Cloneable {

62    private boolean preciseLeakDetectionEnabled = true;
63    private boolean backgroundExpirationEnabled = false;

73    /**
74     * Set the size of the pools we want to configure them with. Pools are
75     * required to control the allocations and deallocations, such that no more
76     * than this number of objects are allocated at any time.
77     *
78     * This means that a pool of size 1, whose single object have expired, will
79     * deallocate that one object before allocating a replacement.
80     *
81     * The size must be at least one for standard pool configurations. A Pool
82     * will throw an {@link IllegalArgumentException} from their constructor if
83     * this is not the case.
84     * @param size The target pool size. Must be at least 1.
85     * @return This Config instance.
86     */
87    public synchronized Config<T> setSize(int size) {
88      this.size = size;
89      return this;
90    }

228   /**
229    * Enable or disable precise object leak detection. It is enabled by default.
230    * Precise object leak detection makes the pool keep an eye on the allocated
231    * Poolables, such that it notices if they get garbage collected without
232    * first being deallocated. Using the garbage collector for this purpose,
233    * means that no false positives (counting objects as leaked, even though
234    * they are not) are ever reported.
235    *
236    * NOTE: While the pool is able to detect object leaks, it cannot prevent
237    * them. All leaks are a sign that there is a bug in the system; most likely
238    * a bug in your code, or in the way the pool is used.
239    *
240    * Precise object leak detection incurs virtually no overhead, and is safe to
241    * leave enabled at all times â€“ even in the most demanding production
242    * environments.
243    *
244    * @param enabled `true` to turn on precise object leak detection (the
245    *                default) `false` to turn it off.
246    * @return This Config instance.
247    */
248   public synchronized Config<T> setPreciseLeakDetectionEnabled(boolean enabled) {
249     this.preciseLeakDetectionEnabled = enabled;
250     return this;
251   }

263   /**
264    * Enable or disable background object expiration checking. This is disabled
265    * by default, because the pool does not know how expensive this check is.
266    * The cost of the check matters because it might end up taking resources
267    * away from the background thread and hinder its ability to keep up with the
268    * demand for allocations and deallocations, even though these tasks always
269    * take priority over any expiration checking.
270    *
271    * @param enabled `true` to turn background expiration checking on, `false`
272    *                (the default) to turn it off.
273    * @return This Config instance.
274    */
275   public synchronized Config<T> setBackgroundExpirationEnabled(boolean enabled) {
276     backgroundExpirationEnabled = enabled;
277     return this;

295    * Check that the configuration is valid in terms of the *standard
296    * configuration*. This method is useful in the
297    * Pool implementation constructors.
298    * @throws IllegalArgumentException If the size is less than one, if the
299    * {@link Expiration} is `null`, if the {@link Allocator} is `null`, or if

302    * provided, because it is really an internal API.

304   @Deprecated // TODO 3.0 make package private
305   public synchronized void validate() throws IllegalArgumentException {
306     if (size < 1) {
307       throw new IllegalArgumentException(
308           "Size must be at least 1, but was " + size);
309     }
310     if (allocator == null) {
311       throw new IllegalArgumentException("Allocator cannot be null");
312     }
313     if (expiration == null) {
314       throw new IllegalArgumentException("Expiration cannot be null");
315     }
316     if (threadFactory == null) {
317       throw new IllegalArgumentException("ThreadFactory cannot be null");
318     }
319   }

321   /**
322    * Returns `null` if no allocator has been configured.
323    * Otherwise returns a `Reallocator`, possibly by adapting the configured
324    * `Allocator` if need be.
325    * If a `MetricsRecorder` has been configured, the return `Reallocator` will
326    * automatically record allocation, reallocation and deallocation latencies.
327    */
328   Reallocator<T> getAdaptedReallocator() {
329     if (allocator == null) {
330       return null;
331     }
332     if (metricsRecorder == null) {
333       if (allocator instanceof Reallocator) {
334         return (Reallocator<T>) allocator;
335       }
336       return new ReallocatingAdaptor<T>((Allocator<T>) allocator);
337     } else {
338       if (allocator instanceof Reallocator) {
339         return new TimingReallocatorAdaptor<T>(
340             (Reallocator<T>) allocator, metricsRecorder);
341       }
342       return new TimingReallocatingAdaptor<T>(
343           (Allocator<T>) allocator, metricsRecorder);
344     }
345   }
