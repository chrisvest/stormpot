16  package stormpot;
17  
18  import java.util.concurrent.TimeUnit;
19  
20  /**
21   * The basic configuration "bean" class.
22   * <p>
23   * Instances of this class is passed to the constructors of {@link Pool pools}
24   * so that they know how big they should be, how to allocate objects and for
25   * how long the objects should live.
26   * <p>
27   * This class is made thread-safe by having the fields be protected by a lock.
28   * The details of this locking mechanism should not be relied upon by client
29   * code.
30   * <p>
31   * The various set* methods are made to return the Config instance itself, so
32   * that the method calls may be chained if so desired.
33   * 
34   * <h3>Standardised configuration</h3>
35   * The contract of the Config class, and how Pools will interpret it, is within
36   * the context of a so-called standardised configuration. All pool and Config
37   * implementations must behave similarly in a standardised configuration.
38   * <p>
39   * It is conceivable that some pool implementations will come with their own
40   * sub-classes of Config, that allow greater control over the pools behaviour.
41   * It is even permissible that these pool that these pool implementations may
42   * deviate from the contract of the Pool interface. However, they are only
43   * allowed to do so in a non-standard configuration. That is, any deviation
44   * from the specified contracts must be explicitly enabled.
45   * 
46   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
47   * @param <T> The type of {@link Poolable} objects that a {@link Pool} based
48   * on this Config will produce.
49   */
50  @SuppressWarnings("unchecked")
51  public class Config<T extends Poolable> {
52  
53    private int size = 10;
54    private long ttl = 10;
55    private TimeUnit ttlUnit = TimeUnit.MINUTES;
56    private Allocator allocator;
57  
58    /**
59     * Set the size of the pools we want to configure them with. Pools are
60     * required to control the allocations and deallocations, such that no more
61     * than this number of objects are allocated at any time.
62     * <p>
63     * This means that a pool of size 1, whose single object have expired, will
64     * deallocate that one object before allocating a replacement.
65     * <p>
66     * The size must be at least one for standard pool configurations. A Pool
67     * will throw an {@link IllegalArgumentException} from their constructor if
68     * this is not the case.
69     * @param size The target pool size. Must be at least 1.
70     * @return This Config instance.
71     */
72    public synchronized Config<T> setSize(int size) {
73      this.size = size;
74      return this;
75    }
76  
77    /**
78     * Get the currently configured size. Default is 10.
79     * @return The configured pool size.
80     */
81    public synchronized int getSize() {
82      return size;
83    }
84  
85    /**
86     * Set the time-to-live for the objects in the pools we want to configure.
87     * Objects older than this value will be reclaimed and their slot
88     * re-allocated.
89     * <p>
90     * Note that no guarantee is given for how much older than the TTL an object
91     * may be, before it is deallocated. Further, the liveness checking is
92     * inherently racy, so objects that are close to the edge of their TTL might
93     * actually have expired by the time they are returned from a claim method.
94     * Pools are permitted to try and account for this by pro-actively
95     * re-allocating objects before their TTL expires, but this behaviour is not
96     * part of the Pool contract and as such should generally not be relied upon. 
97     * <p>
98     * The <code>ttl</code> value must be at least 1 for standard pool
99     * configurations. Further, the <code>unit</code> must be non-null.
100    * A Pool will throw an {@link IllegalArgumentException} from their
101    * constructor if this is not the case.
102    * @param ttl The scalar value of the time-to-live value. Must be at least 1.
103    * @param unit The unit of the 'ttl' value. Cannot be null.
104    * @return This Config instance.
105    */
106   public synchronized Config<T> setTTL(long ttl, TimeUnit unit) {
107     this.ttl = ttl;
108     this.ttlUnit = unit;
109     return this;
110   }
111 
112   /**
113    * Get the scalar value of the time-to-live value that applies to the objects
114    * in the pools we want to configure. Default TTL is 10 minutes.
115    * @return The scalar time-to-live.
116    */
117   public synchronized long getTTL() {
118     return ttl;
119   }
120 
121   /**
122    * Get the time-scale unit of the configured time-to-live that applies to the
123    * objects in the pools we want to configure. Default TTL is 10 minutes.
124    * @return The time-scale unit of the configured time-to-live value.
125    */
126   public synchronized TimeUnit getTTLUnit() {
127     return ttlUnit;
128   }
129 
130   /**
131    * Set the {@link Allocator} to use for the pools we want to configure.
132    * This will change the type-parameter of the Config object to match that
133    * of the new Allocator.
134    * <p>
135    * The allocator is initially <code>null</code> in a new Config object, and
136    * can be set to <code>null</code> any time. However, in a standard
137    * configuration, it must be non-null when the Config is passed to a Pool
138    * constructor. Otherwise the constructor will throw an
139    * {@link IllegalArgumentException}.
140    * @param <X> The type of {@link Poolable} that is created by the allocator,
141    * and the type of objects that the configured pools will contain.
142    * @param allocator The allocator we want our pools to use.
143    * @return This Config instance, but with a generic type parameter that
144    * matches that of the allocator.
145    */
146   public synchronized <X extends Poolable> Config<X> setAllocator(
147       Allocator<X> allocator) {
148     this.allocator = allocator;
149     return (Config<X>) this;
150   }
151 
152   /**
153    * Get the configured {@link Allocator} instance. There is no configured
154    * allocator by default, so this must be {@link #setAllocator(Allocator) set}
155    * before instantiating any Pool implementations from this Config.
156    * @return The configured Allocator instance, if any.
157    */
158   public synchronized Allocator<T> getAllocator() {
159     return allocator;
160   }
161 
162   /**
163    * Configure the Config instance given as a parameter, with all the values
164    * of this Config.
165    * @param config The Config instance whose allocator, size and TTL we want
166    * to configure such that they match our configuration.
167    */
168   public synchronized void setFieldsOn(Config config) {
169     config.setAllocator(allocator);
170     config.setSize(size);
171     config.setTTL(ttl, ttlUnit);
172   }
173 
174   /**
175    * Validate that the configuration is valid. This method is useful in the
176    * Pool implementation constructors.
177    * @throws IllegalArgumentException If the size is less than one, if the TTL
178    * value is less than one, if the TTL {@link TimeUnit} is null, or if the
179    * {@link Allocator} is null.
180    */
181   public synchronized void validate() throws IllegalArgumentException {
182     if (size < 1) {
183       throw new IllegalArgumentException(
184           "size must be at least 1, but was " + size);
185     }
186     if (ttl < 1) {
187       throw new IllegalArgumentException(
188           "TTL value must be at least 1, but was " + ttl);
189     }
190     if (ttlUnit == null) {
191       throw new IllegalArgumentException(
192           "TTL TimeUnit cannot be null");
193     }
194     if (allocator == null) {
195       throw new IllegalArgumentException(
196           "Allocator cannot be null");
197     }
198   }
199 }
