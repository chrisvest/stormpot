16  package stormpot;
17  
18  import java.util.concurrent.TimeUnit;
19  
20  /**
21   * This is the standard time based {@link Expiration}. It will invalidate
22   * objects based on how long ago they were allocated.
23   * 
24   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
25   */
26  public class TimeExpiration implements Expiration<Poolable> {
27  
28    private final long maxPermittedAgeMillis;
29  
30    /**
31     * Construct a new Expiration that will invalidate objects that are older
32     * than the provided span of time in the given unit.
33     * <p>
34     * If the <code>maxPermittedAge</code> is less than one, or the
35     * <code>unit</code> is <code>null</code>, then an
36     * {@link IllegalArgumentException} will be thrown.
37     * 
38     * @param maxPermittedAge Poolables older than this, in the given unit, will
39     * be considered expired. This value must be at least 1.
40     * @param unit The {@link TimeUnit} of the maximum permitted age. Never
41     * <code>null</code>.
42     */
43    public TimeExpiration(long maxPermittedAge, TimeUnit unit) {
44      if (maxPermittedAge < 1) {
45        throw new IllegalArgumentException(
46            "max permitted age cannot be less than 1");
47      }
48      if (unit == null) {
49        throw new IllegalArgumentException("unit cannot be null");
50      }
51      maxPermittedAgeMillis = unit.toMillis(maxPermittedAge);
52    }
53  
54    /**
55     * Returns <code>true</code> if the {@link Poolable} represented by the given
56     * {@link SlotInfo} is older than the maximum age permitted by this
57     * TimeExpiration.
58     */
59    public boolean hasExpired(SlotInfo<? extends Poolable> info) {
60      return info.getAgeMillis() > maxPermittedAgeMillis;
61    }
62  }
