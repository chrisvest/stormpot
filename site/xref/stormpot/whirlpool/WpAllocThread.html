16  package stormpot.whirlpool;
17  
18  import java.util.concurrent.CountDownLatch;
19  import java.util.concurrent.TimeUnit;
20  
21  import stormpot.Allocator;
22  import stormpot.Completion;
23  import stormpot.Config;
24  
25  @SuppressWarnings("unchecked")
26  class WpAllocThread extends Thread implements Completion {
27  
28    private final CountDownLatch shutdownLatch;
29    private final int targetSize;
30    private final Allocator allocator;
31    private final Whirlpool pool;
32    
33    private volatile boolean runnable = true;
34    
35    private int size;
36    private long relieveTimeout = 0;
37    private TimeUnit relieveUnit = TimeUnit.MILLISECONDS;
38  
39    public WpAllocThread(Config config, Whirlpool whirlpool) {
40      super("Whirlpool-Allocator-Thread for " + config.getAllocator());
41      shutdownLatch = new CountDownLatch(1);
42      targetSize = config.getSize();
43      allocator = config.getAllocator();
44      pool = whirlpool;
45    }
46  
47    public void shutdown() {
48      runnable = false;
49    }
50  
51    public void await() throws InterruptedException {
52      shutdownLatch.await();
53    }
54  
55    public boolean await(long timeout, TimeUnit unit) throws InterruptedException {
56      if (unit == null) {
57        throw new IllegalArgumentException("timeout TimeUnit cannot be null");
58      }
59      return shutdownLatch.await(timeout, unit);
60    }
61  
62    @Override
63    public void run() {
64      while (runnable) {
65        reallocate();
66      }
67      // we've been shut down
68      deallocateAll();
69      shutdownLatch.countDown();
70    }
71  
72    private void reallocate() {
73      if (size < targetSize) {
74        WSlot slot = new WSlot(pool, "object");
75        allocate(slot);
76        if (size == targetSize) {
77          relieveTimeout = 10;
78        }
79      }
80      WSlot slot;
81      try {
82        slot = pool.relieve(relieveTimeout, relieveUnit);
83        if (slot != null) {
84          deallocate(slot);
85          allocate(slot);
86        }
87      } catch (InterruptedException e) {
88        // take it as a shutdown signal
89        pool.shutdown();
90      }
91    }
92  
93    private void allocate(WSlot slot) {
94      if (!runnable) {
95        return;
96      }
97      try {
98        slot.obj = allocator.allocate(slot);
99        if (slot.obj != null) {
100         size++;
101       } else {
102         slot.poison = new NullPointerException("allocation returned null");
103       }
104     } catch (Exception e) {
105       slot.poison = e;
106     }
107     slot.created = System.currentTimeMillis();
108     pool.release(slot);
109   }
110 
111   private void deallocate(WSlot slot) {
112     try {
113       if (slot.obj != null) {
114         size--;
115         allocator.deallocate(slot.obj);
116       }
117     } catch (Exception e) {
118       // ignored as per spec.
119     }
120     slot.created = 0;
121     slot.next = null;
122     slot.obj = null;
123     slot.poison = null;
124     slot.claimed = false;
125   }
126 
127   private void deallocateAll() {
128     while (size > 0) {
129       WSlot slot;
130       try {
131         slot = pool.relieve(relieveTimeout, relieveUnit);
132         if (slot == null) {
133           continue;
134         }
135         deallocate(slot);
136       } catch (InterruptedException e) {
137         // ignore interrupts - we're already trying to shut down
138       }
139     }
140   }
141 }
