29  /**
30   * The `PoolBuilder` collects information about how big a pool should be,
31   * and how it should allocate objects, an so on, and finally acts as the
32   * factory for building the pool instances themselves, with the
33   * {@link #build()} method.
34   *
35   * Pool builder instances are obtained by calling one of the `from*`
36   * methods on {@link Pool}, such as {@link Pool#fromThreaded(Allocator)}.
37   *
38   * This class is made thread-safe by having the fields be protected by the
39   * intrinsic object lock on the `PoolBuilder` object itself. This way, pools
40   * can `synchronize` on the builder object to read the values out atomically.
41   *
42   * The various set* methods are made to return the `PoolBuilder` instance
43   * itself, so that the method calls may be chained if so desired.
44   * 
45   * @author Chris Vest <mr.chrisvest@gmail.com>
46   * @param <T> The type of {@link Poolable} objects that a {@link Pool} based
47   * on this `PoolBuilder` will produce.
48   */
49  public final class PoolBuilder<T extends Poolable> implements Cloneable {
50    static final Map<AllocationProcessMode, PoolBuilderDefaults> DEFAULTS = Map.of(
51        THREADED, new PoolBuilderDefaults(after(8, 10, MINUTES), INSTANCE, true, true, 1000),
52        INLINE, new PoolBuilderDefaults(after(8, 10, MINUTES), INSTANCE, true, false, 0),
53        DIRECT, new PoolBuilderDefaults(never(), INSTANCE, false, false, 0)
54    );
55  
56    static final Map<AllocationProcessMode, PoolBuilderPermissions> PERMISSIONS = Map.of(
57        THREADED, new PoolBuilderPermissions(true, true, true, true, true),
58        INLINE, new PoolBuilderPermissions(true, true, true, false, false),
59        DIRECT, new PoolBuilderPermissions(false, true, false, false, false)
60    );
61  
62    private final AllocationProcess allocationProcess;
63    private final PoolBuilderPermissions permissions;
64    private Allocator<T> allocator;
65    private int size = 10;
66    private Expiration<? super T> expiration;
67    private MetricsRecorder metricsRecorder;
68    private ThreadFactory threadFactory;
69    private boolean preciseLeakDetectionEnabled;
70    private boolean backgroundExpirationEnabled;
71    private int backgroundExpirationCheckDelay;
72  
73    /**
74     * Build a new empty `PoolBuilder` object.
75     */
76    PoolBuilder(AllocationProcess allocationProcess, Allocator<T> allocator) {
77      requireNonNull(allocator, "The Allocator cannot be null.");
78      requireNonNull(allocationProcess, "The AllocationProcess cannot be null.");
79      this.allocator = allocator;
80      this.allocationProcess = allocationProcess;
81      this.permissions = PERMISSIONS.get(allocationProcess.getMode());
82      PoolBuilderDefaults defaults = DEFAULTS.get(allocationProcess.getMode());
83      this.expiration = defaults.expiration;
84      this.threadFactory = defaults.threadFactory;
85      this.preciseLeakDetectionEnabled = defaults.preciseLeakDetectionEnabled;
86      this.backgroundExpirationEnabled = defaults.backgroundExpirationEnabled;
87      this.backgroundExpirationCheckDelay = defaults.backgroundExpirationCheckDelay;
88    }

90    /**
91     * Set the size of the pool we are building.
92     *
93     * Pools are required to control the allocations and deallocations, such that
94     * no more than this number of objects are allocated at any time.
95     *
96     * This means that a pool of size 1, whose single object have expired, will
97     * deallocate that one object before allocating a replacement.
98     *
99     * The size must be at least one, or an {@link IllegalArgumentException} will
100    * be thrown when building the pool.
101    *
102    * Note that the pool size can be modified after the pool has been built, by
103    * calling the {@link Pool#setTargetSize(int)} or
104    * {@link ManagedPool#setTargetSize(int)} methods.
105    *
106    * @param size The target pool size. Must be at least 0.
107    * @return This `PoolBuilder` instance.
108    */
109   public synchronized PoolBuilder<T> setSize(int size) {
110     checkPermission(permissions.setSize, "size");
111     if (size < 0) {
112       throw new IllegalArgumentException("Size must be at least 0, but was " + size + ".");
113     }
114     this.size = size;
115     return this;
116   }
117 
118   /**
119    * Get the currently configured size. The default is 10.
120    * @return The configured pool size.
121    */
122   public synchronized int getSize() {
123     return size;
124   }

126   /**
127    * Set the {@link Allocator} or {@link Reallocator} to use for the pools we
128    * want to configure. This will change the type-parameter of the `PoolBuilder`
129    * object to match that of the new Allocator.
130    *
131    * The allocator is initially specified by the {@link Pool#from(Allocator)}
132    * method, so there is usually no need to set it later.
133    *
134    * @param allocator The allocator we want our pools to use.
135    * This cannot be `null`.
136    * @param <X> The type of {@link Poolable} that is created by the allocator,
137    * and the type of objects that the configured pools will contain.
138    * @return This `PoolBuilder` instance, but with a generic type parameter that
139    * matches that of the allocator.
140    */
141   @SuppressWarnings("unchecked")
142   public synchronized <X extends Poolable> PoolBuilder<X> setAllocator(
143       Allocator<X> allocator) {
144     checkPermission(permissions.setAllocator, "allocator");
145     requireNonNull(allocator, "The Allocator cannot be null.");
146     this.allocator = (Allocator<T>) allocator;
147     return (PoolBuilder<X>) this;
148   }

151    * Get the configured {@link Allocator} instance.
152    * @return The configured Allocator instance.

155     return allocator;
156   }

158   /**
159    * Get the configured {@link stormpot.Allocator} instance as a
160    * {@link stormpot.Reallocator}. If the configured allocator implements the
161    * Reallocator interface, then it is returned directly. Otherwise, the
162    * allocator is wrapped in an adaptor.
163    * @return A configured or adapted Reallocator.
164    */
165   public synchronized Reallocator<T> getReallocator() {
166     if (allocator instanceof Reallocator) {
167       return (Reallocator<T>) allocator;
168     }
169     return new ReallocatingAdaptor<>(allocator);
170   }

172   /**
173    * Set the {@link Expiration} to use for the pools we want to
174    * configure. The Expiration determines when a pooled object is valid

176    *
177    * The default Expiration is an
178    * {@link Expiration#after(long, long, TimeUnit)} that invalidates the
179    * objects after they have been active for somewhere between 8 to 10 minutes.
180    *
181    * @param expiration The expiration we want our pools to use. Not null.
182    * @return This `PoolBuilder` instance.
183    */
184   public synchronized PoolBuilder<T> setExpiration(Expiration<? super T> expiration) {
185     checkPermission(permissions.setExpiration, "expiration");
186     requireNonNull(expiration, "Expiration cannot be null.");
187     this.expiration = expiration;
188     return this;
189   }

248   /**
249    * Return whether or not precise object leak detection is enabled, which is
250    * the case by default.
251    * @return `true` if precise object leak detection is enabled.
252    * @see #setPreciseLeakDetectionEnabled(boolean)
253    */
254   public synchronized boolean isPreciseLeakDetectionEnabled() {
255     return preciseLeakDetectionEnabled;
256   }
257 
258   /**
259    * Enable or disable precise object leak detection. It is enabled by default.
260    * Precise object leak detection makes the pool keep an eye on the allocated
261    * Poolables, such that it notices if they get garbage collected without
262    * first being deallocated. Using the garbage collector for this purpose,
263    * means that no false positives (counting objects as leaked, even though
264    * they are not) are ever reported.
265    *
266    * [NOTE]
267    * --
268    * While the pool is able to detect object leaks, it cannot prevent
269    * them. All leaks are a sign that there is a bug in the system; most likely
270    * a bug in your code, or in the way the pool is used.
271    * --
272    *
273    * Precise object leak detection incurs virtually no overhead, and is safe to
274    * leave enabled at all times – even in the most demanding production
275    * environments.
276    *
277    * @param enabled `true` to turn on precise object leak detection (the
278    *                default) `false` to turn it off.
279    * @return This `PoolBuilder` instance.
280    */
281   public synchronized PoolBuilder<T> setPreciseLeakDetectionEnabled(boolean enabled) {
282     this.preciseLeakDetectionEnabled = enabled;
283     return this;
284   }

286   /**
287    * Return whether or not background expiration is enabled.
288    * By default, background expiration is enabled.
289    *
290    * @return `true` if background expiration is enabled.
291    * @see #setBackgroundExpirationEnabled(boolean)
292    */
293   public synchronized boolean isBackgroundExpirationEnabled() {
294     return backgroundExpirationEnabled;
295   }

297   /**
298    * Enable or disable background object expiration checking. This is enabled
299    * by default, but can be turned off if the check is expensive.
300    * The cost of the check matters because it might end up taking resources
301    * away from the background thread and hinder its ability to keep up with the
302    * demand for allocations and deallocations, even though these tasks always
303    * take priority over any expiration checking.
304    *
305    * @param enabled `true` (the default) to turn background expiration checking on,
306    *               `false` to turn it off.
307    * @return This `PoolBuilder` instance.
308    */
309   public synchronized PoolBuilder<T> setBackgroundExpirationEnabled(boolean enabled) {
310     checkPermission(permissions.setBackgroundExpiration, "background expiration enabled/disabled");
311     backgroundExpirationEnabled = enabled;
312     return this;
313   }

315   /**
316    * Return the default approximate delay, in milliseconds, between background
317    * maintenance tasks, such as the background expiration checks and retrying
318    * failed allocations.
319    *
320    * @return the delay, in milliseconds, between background maintenance tasks.
321    */
322   public synchronized int getBackgroundExpirationCheckDelay() {
323     return backgroundExpirationCheckDelay;
324   }
325 
326   /**
327    * Set the approximate delay, in milliseconds, between background maintenance tasks.
328    * These tasks include the background expiration checks, and retrying failed allocations.
329    *
330    * The default delay is 1.000 milliseconds (1 second). Lowering this value will
331    * improve the pools responsiveness to repairing failed allocations, and also increase
332    * the frequency of the background expiration checks. This comes at the cost of higher
333    * idle CPU usage.
334    *
335    * It is not recommended to set this value lower than 100 milliseconds. Values lower
336    * than this tend to have increased CPU and power usage, for very little gain in
337    * responsiveness for the background tasks.
338    *
339    * @param delay the desired delay, in milliseconds, between background maintenance tasks.
340    * @return This `PoolBuilder` instance.
341    */
342   public synchronized PoolBuilder<T> setBackgroundExpirationCheckDelay(int delay) {
343     checkPermission(permissions.setBackgroundExpiration, "background expiration check delay");
344     if (delay < 0) {
345       throw new IllegalArgumentException("Background expiration check delay cannot be negative.");
346     }
347     backgroundExpirationCheckDelay = delay;
348     return this;
349   }

366   /**
367    * Build a {@link Pool} instance based on the collected configuration.
368    * @return A {@link Pool} instance as configured by this builder.
369    */
370   public synchronized Pool<T> build() {
371     return new BlazePool<>(this, allocationProcess);
372   }
373 
374   /**
375    * Returns a `Reallocator`, possibly by adapting the configured
376    * `Allocator` if need be.
377    * If a `MetricsRecorder` has been configured, the return `Reallocator` will
378    * automatically record allocation, reallocation and deallocation latencies.
379    */
380   synchronized Reallocator<T> getAdaptedReallocator() {
381     if (metricsRecorder == null) {
382       if (allocator instanceof Reallocator) {
383         return (Reallocator<T>) allocator;
384       }
385       return new ReallocatingAdaptor<>(allocator);
386     } else {
387       if (allocator instanceof Reallocator) {
388         return new TimingReallocatorAdaptor<>(
389             (Reallocator<T>) allocator, metricsRecorder);
390       }
391       return new TimingReallocatingAdaptor<>(allocator, metricsRecorder);
392     }
393   }
394 
395   private void checkPermission(boolean permission, String fieldDescription) {
396     if (!permission) {
397       String message = "The " + allocationProcess.getMode() + " allocation process does " +
398           "not support configuring the " + fieldDescription + ".";
399       throw new IllegalStateException(message);
400     }
401   }
