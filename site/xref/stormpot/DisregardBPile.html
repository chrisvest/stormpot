16  package stormpot;
17  
18  import java.util.concurrent.BlockingQueue;
19  import java.util.concurrent.atomic.AtomicReference;
20  
21  @SuppressWarnings("unchecked")
22  final class DisregardBPile<T extends Poolable>
23      extends AtomicReference<DisregardedBSlot<T>> {
24    private static final long serialVersionUID = 2374582348576873465L;
25    private static final DisregardedBSlot<Poolable> STACK_END =
26        new DisregardedBSlot<>(null);
27  
28    private final BlockingQueue<BSlot<T>> refillQueue;
29  
30    DisregardBPile(BlockingQueue<BSlot<T>> refillQueue) {
31      this.refillQueue = refillQueue;
32      set((DisregardedBSlot<T>) STACK_END);
33    }
34  
35    void addSlot(BSlot<T> slot) {
36      DisregardedBSlot<T> element = new DisregardedBSlot<>(slot);
37      element.next = getAndSet(element);
38    }
39  
40    boolean refillQueue() {
41      DisregardedBSlot<T> stack = getAndSet((DisregardedBSlot<T>) STACK_END);
42      int count = 0;
43      while (stack != STACK_END) {
44        count++;
45        refillQueue.offer(stack.slot);
46        DisregardedBSlot<T> next;
47        do {
48          next = stack.next;
49        } while (next == null);
50        stack = next;
51      }
52      return count > 0;
53    }
54  }
