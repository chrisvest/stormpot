16  package stormpot;
17  
18  import java.util.Objects;
19  import java.util.concurrent.ThreadLocalRandom;
20  import java.util.concurrent.TimeUnit;
21  
22  /**
23   * This is the standard time based {@link Expiration}. It will invalidate
24   * objects based on about how long ago they were allocated.
25   *
26   * @author Chris Vest <mr.chrisvest@gmail.com>
27   * @since 2.2
28   */
29  final class TimeSpreadExpiration<T extends Poolable> implements Expiration<T> {
30  
31    private final long fromMillis;
32    private final long toMillis;
33    private final TimeUnit unit;
34  
35    /**
36     * Construct a new Expiration that will invalidate objects that are older
37     * than the given lower bound, before they get older than the upper bound,
38     * in the given time unit.
39     *
40     * If the `lowerBound` is less than 1, the `upperBound` is less than the
41     * `lowerBound`, or the `unit` is `null`, then an
42     * {@link java.lang.IllegalArgumentException} will be thrown.
43     *
44     * @param lowerBound Poolables younger than this, in the given unit, are not
45     *                   considered expired. This value must be at least 1.
46     * @param upperBound Poolables older than this, in the given unit, are always
47     *                   considered expired. This value must be greater than the
48     *                   lowerBound.
49     * @param unit The {@link TimeUnit} of the bounds values. Never `null`.
50     */
51    TimeSpreadExpiration(
52        long lowerBound,
53        long upperBound,
54        TimeUnit unit) {
55      Objects.requireNonNull(unit, "TimeUnit cannot be null.");
56      if (lowerBound < 1) {
57        throw new IllegalArgumentException(
58            "The lower bound cannot be less than 1.");
59      }
60      if (upperBound < lowerBound) {
61        throw new IllegalArgumentException(
62            "The upper bound must be greater than or equal to the lower bound.");
63      }
64      this.fromMillis = unit.toMillis(lowerBound);
65      this.toMillis = unit.toMillis(upperBound);
66      this.unit = unit;
67    }
68  
69    /**
70     * Returns `true`, with uniformly increasing probability, if the
71     * {@link stormpot.Poolable} represented by the given {@link SlotInfo} is
72     * older than the lower bound, eventually returning `true` with
73     * 100% certainty when the age of the Poolable is equal to or greater than
74     * the upper bound.
75     *
76     * The uniformity of the random expiration holds regardless of how often a
77     * Poolable is checked. That is to say, checking a Poolable more times within
78     * an interval of time, does _not_ increase its chances of being
79     * declared expired.
80     */
81    @Override
82    public boolean hasExpired(SlotInfo<? extends T> info) {
83      long expirationAge = info.getStamp();
84      if (expirationAge == 0) {
85        expirationAge = computeExpirationDeadline();
86        info.setStamp(expirationAge);
87      }
88      long age = info.getAgeMillis();
89      return age >= expirationAge;
90    }
91  
92    long computeExpirationDeadline() {
93      long maxDelta = toMillis - fromMillis;
94      ThreadLocalRandom prng = ThreadLocalRandom.current();
95      long fudgeFactor = fromMillis == toMillis ? 0 : Math.abs(prng.nextInt() % maxDelta);
96      return fromMillis + fudgeFactor;
97    }
98  
99    /**
100    * Produces a String representation of this TimeSpreadExpiration.
101    */
102   @Override
103   public String toString() {
104     long lower = unit.convert(fromMillis, TimeUnit.MILLISECONDS);
105     long upper = unit.convert(toMillis, TimeUnit.MILLISECONDS);
106     return "TimeSpreadExpiration(" + lower + " to " + upper + " " + unit + ")";
107   }
108 }
