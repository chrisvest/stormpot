16  package stormpot;
17  
18  import java.lang.reflect.Constructor;
19  import java.util.concurrent.BlockingQueue;
20  import java.util.concurrent.LinkedBlockingQueue;
21  
22  final class QueueFactory {
23    interface Factory {
24      <T> BlockingQueue<T> createUnboundedBlockingQueue();
25    }
26  
27    static enum FactoryChoiceReason {
28      DEFAULT_FIRST_CHOICE("The " + QUEUE_IMPL_FIRST_CHOICE + " was available on the classpath," +
29              " and the " + QUEUE_IMPL + " system property did not specify any alternative" +
30              " implementation to use."),
31      SPECIFIED_CHOICE("The " + QUEUE_IMPL + " system property specified the queue implementation" +
32              " to use, specifically " + System.getProperty(QUEUE_IMPL) + "."),
33      DEFAULT_SECOND_CHOICE("The " + QUEUE_IMPL_FIRST_CHOICE + " was not available on the" +
34              " classpath, and the " + QUEUE_IMPL + " system property did not specify any" +
35              " alternative implementation to use, so " + QUEUE_IMPL_SECOND_CHOICE + " was used" +
36              " as the default fallback."),
37      DEFAULT_FALLBACK("")
38      ;
39  
40      private final String description;
41  
42      FactoryChoiceReason(String description) {
43        this.description = description;
44      }
45  
46      public String toString() {
47        return super.toString() + "(" + description + ")";
48      }
49    }
50  
51    private static final Factory factory;
52    private static final FactoryChoiceReason reason;
53  
54    // stop checking line length
55    private static final String QUEUE_IMPL = "stormpot.blocking.queue.impl";
56    private static final String QUEUE_IMPL_FIRST_CHOICE = "java.util.concurrent.LinkedTransferQueue";
57    private static final String QUEUE_IMPL_SECOND_CHOICE = LinkedBlockingQueue.class.getCanonicalName();
58    // start checking line length
59  
60    static {
61      Factory theFactory = null;
62      FactoryChoiceReason theReason = FactoryChoiceReason.SPECIFIED_CHOICE;
63      String queueType = System.getProperty(QUEUE_IMPL);
64  
65      if (queueType == null) {
66        queueType = QUEUE_IMPL_FIRST_CHOICE;
67        theReason = FactoryChoiceReason.DEFAULT_FIRST_CHOICE;
68      }
69  
70      try {
71        Class<?> queueClass = Class.forName(queueType);
72        final Constructor<?> constructor = queueClass.getConstructor();
73        BlockingQueue.class.cast(constructor.newInstance()); // Just making sure that it works...
74        theFactory = new Factory() {
75          @SuppressWarnings("unchecked")
76          @Override
77          public <T> BlockingQueue<T> createUnboundedBlockingQueue() {
78            try {
79              return (BlockingQueue<T>) constructor.newInstance();
80            } catch (Exception e) {
81              Exception exception = new Exception(
82                  "WARNING: Expected class to be constructable: " + constructor,
83                  e);
84              exception.printStackTrace();
85            }
86            return new LinkedBlockingQueue<T>();
87          }
88        };
89      } catch (Exception e) {
90        if (theReason == FactoryChoiceReason.SPECIFIED_CHOICE) {
91          Exception exception = new Exception(
92              "WARNING: The queue implementation specified by the " + QUEUE_IMPL + " system" +
93              " property, specifically " + queueType + ", was determined to be unusable." +
94              " Falling back to the " + QUEUE_IMPL_SECOND_CHOICE + " implementation.",
95              e);
96          exception.printStackTrace();
97        }
98      }
99  
100     if (theFactory == null) {
101       theFactory = new Factory() {
102         @Override
103         public <T> BlockingQueue<T> createUnboundedBlockingQueue() {
104           return new LinkedBlockingQueue<T>();
105         }
106       };
107       theReason = theReason == FactoryChoiceReason.DEFAULT_FIRST_CHOICE?
108           FactoryChoiceReason.DEFAULT_SECOND_CHOICE : FactoryChoiceReason.DEFAULT_FALLBACK;
109     }
110 
111     factory = theFactory;
112     reason = theReason;
113   }
114 
115   static <T> BlockingQueue<T> createUnboundedBlockingQueue() {
116     return factory.createUnboundedBlockingQueue();
117   }
118 
119   static FactoryChoiceReason getQueueFactoryChoiceReason() {
120     return reason;
121   }
122 }
