16  package stormpot;
17  
18  import javax.management.MXBean;
19  
20  /**
21   * This is the JMX management interface for Stormpot object pools.
22   *
23   * Using this interface, pools can be exposed to external management as an
24   * MXBean. Since its an MXBean, and not just an MBean, it imposes no
25   * requirement that external parties knows about the Stormpot types.
26   *
27   * Once you have created your pool, it is easy to expose it through the platform
28   * MBeanServer, or any MBeanServer you like:
29   *
30   * [source,java]
31   * --
32   * BlazePool<MyPoolable> pool = new BlazePool<MyPoolable>(...);
33   * MBeanServer server = ManagementFactory.getPlatformMBeanServer();
34   * ObjectName name = new ObjectName("com.myapp:objectpool=stormpot");
35   * server.registerMBean(pool, name);
36   * --
37   *
38   * Using the platform MBeanServer will make the pool visible to tools like
39   * JConsole and VisualVM.
40   * @since 2.3
41   */
42  @MXBean
43  public interface ManagedPool {
44    /**
45     * Return the number of objects the pool has allocated since it was created.
46     * @return The number of Poolable objects ever created by this pool.
47     */
48    long getAllocationCount();
49  
50    /**
51     * Return the number of allocations that has failed, either because the
52     * allocator threw an exception or because it returned null, since the pool
53     * was created.
54     * @return The number of allocations that have failed for one reason or
55     * another.
56     */
57    long getFailedAllocationCount();
58  
59    /**
60     * If the pool is capable of precise object leak detection, this method will
61     * return the number of object leaks that have been detected, and prevented,
62     * since the pool was created. If the pool does not support precise object
63     * leak detection, then this method returns -1.
64     *
65     * There are two kinds of leaks: One where the application forgets to release
66     * an object back to the pool, but keeps a strong reference to the object,
67     * and another where the application not only forgets to release the object,
68     * but also looses the reference to the object, making it eligible for
69     * garbage collection. The precise leak detector will only count leaks of the
70     * latter kind, where the leaked object has been garbage collected.
71     *
72     * @return The number of objects leaked from the users of this pool, since
73     * the pool was created, or -1 if the pool does not implement precise leak
74     * detection.
75     */
76    long getLeakedObjectsCount();
77  
78    // The *TargetSize methods are duplicated here, because we would otherwise
79    // have to provide a type parameter for the ResizablePool interface, if we
80    // were to extend it.
81  
82    /**
83     * @param size The new target size.
84     * @see ResizablePool#setTargetSize(int)
85     */
86    void setTargetSize(int size);
87  
88    /**
89     * @return The current target size.
90     * @see ResizablePool#getTargetSize()
91     */
92    int getTargetSize();
93  
94    /**
95     * Returns 'true' if the shut down process has been started on this pool,
96     * 'false' otherwise. This method does not reveal whether or not the shut
97     * down process has completed.
98     * @return 'true' if {@link LifecycledPool#shutdown()} has been called on
99     * this pool.
100    */
101   boolean isShutDown();
102 
103   /**
104    * @see stormpot.MetricsRecorder#getObjectLifetimePercentile(double)
105    * @param percentile The percentile to get, as a decimal, e.g. a number
106    *                   between 0.0 and 1.0.
107    * @return The approximate object lifetime in milliseconds, for the given
108    * percentile, or Double.NaN if no {@link stormpot.MetricsRecorder} has been
109    * configured for the pool.
110    */
111   double getObjectLifetimePercentile(double percentile);
112 
113   /**
114    * @see stormpot.MetricsRecorder#getAllocationLatencyPercentile(double)
115    * @param percentile The percentile to get, as a decimal, e.g. a number
116    *                   between 0.0 and 1.0.
117    * @return The approximate latency for allocations in milliseconds, for the
118    * given percentile, or Double.NaN if no {@link stormpot.MetricsRecorder}
119    * has been configured for the pool.
120    */
121   double getAllocationLatencyPercentile(double percentile);
122 
123   /**
124    * @see stormpot.MetricsRecorder#getAllocationFailureLatencyPercentile(double)
125    * @param percentile The percentile to get, as a decimal, e.g. a number
126    *                   between 0.0 and 1.0.
127    * @return The approximate latency for failed allocations in milliseconds,
128    * for the given percentile, or Double.NaN if no
129    * {@link stormpot.MetricsRecorder} has been configured for the pool.
130    */
131   double getAllocationFailureLatencyPercentile(double percentile);
132 
133   /**
134    * @see stormpot.MetricsRecorder#getReallocationLatencyPercentile(double)
135    * @param percentile The percentile to get, as a decimal, e.g. a number
136    *                   between 0.0 and 1.0.
137    * @return The approximate latency for reallocations in milliseconds, for the
138    * given percentile, or Double.NaN if no
139    * {@link stormpot.MetricsRecorder} has been configured for the pool.
140    */
141   double getReallocationLatencyPercentile(double percentile);
142 
143   /**
144    * @see stormpot.MetricsRecorder#getReallocationFailurePercentile(double)
145    * @param percentile The percentile to get, as a decimal, e.g. a number
146    *                   between 0.0 and 1.0.
147    * @return The approximate latency for failed reallocations in milliseconds,
148    * for the given percentile, or Double.NaN if no
149    * {@link stormpot.MetricsRecorder} has been configured for the pool.
150    */
151   double getReallocationFailureLatencyPercentile(double percentile);
152 
153   /**
154    * @see stormpot.MetricsRecorder#getDeallocationLatencyPercentile(double)
155    * @param percentile The percentile to get, as a decimal, e.g. a number
156    *                   between 0.0 and 1.0.
157    * @return The approximate latency for deallocations, regardless of whether
158    * the throw exceptions or not, in milliseconds for the given percentile, or
159    * Double.NaN if no {@link stormpot.MetricsRecorder} has been configured for
160    * the pool.
161    */
162   double getDeallocationLatencyPercentile(double percentile);
163 }
