16  package stormpot.bpool;

18  import java.util.concurrent.BlockingQueue;
19  import java.util.concurrent.atomic.AtomicInteger;
20  
21  import stormpot.PoolException;
22  import stormpot.Poolable;
23  import stormpot.Slot;
24  import stormpot.SlotInfo;
25  
26  class BSlot<T extends Poolable> implements Slot, SlotInfo<T> {
27    static final int LIVING = 1;
28    static final int CLAIMED = 2;
29    static final int TLR_CLAIMED = 3;
30    static final int DEAD = 4;
31    
32    final BlockingQueue<BSlot<T>> live;
33    // TODO make BSlot extend AtomicInt instead, to avoid the indirection.
34    private final AtomicInteger state;
35    T obj;
36    Exception poison;
37    long created;
38    long claims;
39    
40    public BSlot(BlockingQueue<BSlot<T>> live) {
41      this.live = live;
42      this.state = new AtomicInteger(DEAD);
43    }
44    
45    public void release(Poolable obj) {
46      int slotState = 0;
47      do {
48        slotState = state.get();
49        // We loop here because TLR_CLAIMED slots can be concurrently changed
50        // into normal CLAIMED slots.
51      } while (!releaseState(slotState));
52      if (slotState == CLAIMED) {
53        live.offer(this);
54      }
55    }
56    
57    private boolean releaseState(int slotState) {
58      if (slotState == TLR_CLAIMED) {
59        return claimTlr2live();
60      } else if (slotState == CLAIMED) {
61        return claim2live();
62      }
63      throw new PoolException("Slot release from bad state: " + slotState);
64    }
65    
66    public boolean claim2live() {
67      // why would this ever fail?
68      return cas(CLAIMED, LIVING);
69      
70      // TODO maybe we can do this instead?
71  //    state.lazySet(LIVING);
72  //    return true;
73    }
74    
75    public boolean claimTlr2live() {
76      return cas(TLR_CLAIMED, LIVING);
77    }
78    
79    public boolean live2claim() {
80      return cas(LIVING, CLAIMED);
81    }
82    
83    public boolean live2claimTlr() {
84      return cas(LIVING, TLR_CLAIMED);
85    }
86    
87    public boolean claimTlr2claim() {
88      return cas(TLR_CLAIMED, CLAIMED);
89    }
90    
91    public boolean claim2dead() {
92      return cas(CLAIMED, DEAD);
93    }
94    
95    public boolean dead2live() {
96      return cas(DEAD, LIVING);
97    }
98    
99    public boolean live2dead() {
100     return cas(LIVING, DEAD);
101   }
102

116   @Override
117   public long getAgeMillis() {
118     return System.currentTimeMillis() - created;
119   }
120 
121   @Override
122   public long getClaimCount() {
123     return claims;
124   }
125 
126   @Override
127   public T getPoolable() {
128     return obj;
129   }
130 
131   public boolean isDead() {
132     return state.get() == DEAD;
133   }
134   
135   public int getState() {
136     return state.get();
137   }
