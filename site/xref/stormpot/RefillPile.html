16  package stormpot;
17  
18  import java.util.concurrent.BlockingQueue;
19  import java.util.concurrent.atomic.AtomicReference;
20  
21  /**
22   * A `RefillPile` can collect objects, in a concurrent and wait-free manner,
23   * before releasing them all to a queue.
24   */
25  @SuppressWarnings("unchecked")
26  final class RefillPile<T extends Poolable>
27      extends AtomicReference<RefillSlot<T>> {
28    private static final long serialVersionUID = 2374582348576873465L;
29    private static final RefillSlot<Poolable> STACK_END =
30        new RefillSlot<>(null);
31  
32    private final BlockingQueue<BSlot<T>> refillQueue;
33  
34    RefillPile(BlockingQueue<BSlot<T>> refillQueue) {
35      this.refillQueue = refillQueue;
36      set((RefillSlot<T>) STACK_END);
37    }
38  
39    /**
40     * Push the given slot onto the stack. This method is wait-free.
41     * @param slot The slot instance to be pushed onto the stack.
42     */
43    void push(BSlot<T> slot) {
44      RefillSlot<T> element = new RefillSlot<>(slot);
45      element.next = getAndSet(element);
46    }
47  
48    BSlot<T> pop() {
49      RefillSlot<T> element;
50      RefillSlot<T> next;
51      do {
52        element = get();
53        if (element == STACK_END) {
54          return null;
55        }
56        next = element.next;
57      } while ((next == null && pause()) || !compareAndSet(element, next));
58      return element.slot;
59    }
60  
61    private boolean pause() {
62      Thread.onSpinWait();
63      return true;
64    }
65  
66    /**
67     * Refill the target queue with all the slots that have been pushed onto this stack.
68     * This method atomically pops all elements from the stack at once, and then pushed onto the
69     * queue one by one.
70     * @return `true` if any slots has been offered to the queue, or `false` if there were no
71     * slots in the pile.
72     */
73    boolean refill() {
74      RefillSlot<T> stack = getAndSet((RefillSlot<T>) STACK_END);
75      int count = 0;
76      while (stack != STACK_END) {
77        count++;
78        refillQueue.offer(stack.slot);
79        RefillSlot<T> next;
80        do {
81          next = stack.next;
82        } while (next == null);
83        stack = next;
84      }
85      return count > 0;
86    }
87  
88    @Override
89    public String toString() {
90      RefillSlot<T> currentValue = get();
91      return "RefillPile[" + (currentValue == STACK_END ? "EMPTY" : currentValue) + "]";
92    }
93  }
