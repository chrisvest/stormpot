16  package stormpot;
17  
18  import java.time.Duration;
19  import java.util.Objects;
20  import java.util.concurrent.TimeUnit;
21  
22  /**
23   * A Timeout represents the maximum amount of time a caller is willing to wait
24   * for a blocking operation to complete.
25   *
26   * Timeouts are independent of their units, so two timeouts of equivalent
27   * duration but constructed in different units, will be equal to each other and
28   * work exactly the same.
29   *
30   * Timeouts are also independent of "calendar time" in the sense that they
31   * represent and work as a duration of absolute time. In other words, timeouts
32   * do not grow or shrink with passing leap seconds or daylight savings time
33   * adjustments.
34   *
35   * @author Chris Vest <mr.chrisvest@gmail.com>
36   */
37  public final class Timeout {
38    private final long timeout;
39    private final TimeUnit unit;
40    private final long timeoutBase;
41    
42    /**
43     * Construct a new timeout with the given value and unit. The unit cannot be
44     * `null`, but the timeout value is unrestricted. The meaning of a
45     * negative timeout value is specific to the implementation of the use site,
46     * but typically means that no amount of blocking or waiting is tolerated.
47     * @param timeout A numerical value for the timeout. Can be zero or negative,
48     * though the meaning is implementation specific.
49     * @param unit The unit of the timeout value. Never `null`.
50     */
51    public Timeout(long timeout, TimeUnit unit) {
52      Objects.requireNonNull(unit, "The TimeUnit cannot be null.");
53      this.timeout = timeout;
54      this.unit = unit;
55      this.timeoutBase = getBaseUnit().convert(timeout, unit);
56    }
57  
58    /**
59     * Construct a new timeout with the given duration.
60     * An exception will be thrown if the duration cannot be converted to a nanosecond quantity.
61     * The duration also cannot be `null`.
62     * Apart from these two, there are no restrictions on the duration.
63     * @param duration A duration to use for the timeout.
64     */
65    public Timeout(Duration duration) {
66      Objects.requireNonNull(duration, "Duration cannot be null.");
67      this.timeout = duration.toNanos();
68      this.unit = TimeUnit.NANOSECONDS;
69      this.timeoutBase = timeout;
70    }
71  
72    /**
73     * Get the timeout value in terms of the {@link #getUnit() unit}.
74     * @return A numerical value of the timeout. Possibly zero or negative.
75     */
76    public long getTimeout() {
77      return timeout;
78    }
79  
80    /**
81     * Get the unit for the {@link #getTimeout() timeout value}.
82     * @return The {@link TimeUnit} of the timeout value. Never `null`.
83     */
84    public TimeUnit getUnit() {
85      return unit;
86    }
87  
88    /**
89     * Calculate a deadline, as an instant in the future, in terms of the
90     * {@link #getBaseUnit() base unit}. Once you have a deadline, you can ask
91     * how much time is left until it transpires, with the
92     * {@link #getTimeLeft(long)} method, giving the deadline as an argument.
93     *
94     * If the {@link #getTimeout() timeout value} is really small, zero or
95     * negative, then the deadline might be an instant in the past.
96     * @return A numerical value that represents the deadline from "now" until
97     * this timeout has passed.
98     *
99     * @deprecated This method is deprecated because it encourages code that is
100    * susceptible to overflow bugs.
101    */
102   @SuppressWarnings("WeakerAccess")
103   @Deprecated(forRemoval = true)
104   public long getDeadline() {
105     return NanoClock.nanoTime() + timeoutBase;
106   }
107 
108   /**
109    * Get the timeout value in terms of the {@link #getBaseUnit() base unit}.
110    * @return A numerical value of the timeout. Possibly zero or negative.
111    */
112   public long getTimeoutInBaseUnit() {
113     return timeoutBase;
114   }
115 
116   /**
117    * Calculate the amount of time that is left until the deadline, in terms
118    * of the {@link #getBaseUnit() base unit}. The argument is a deadline value
119    * that has been calculated with the {@link #getDeadline()} method.
120    * @param deadline The reference deadline from {@link #getDeadline()}.
121    * @return The amount of time, in terms of the
122    * {@link #getBaseUnit() base unit}, that is left until the deadline. If
123    * this value is negative, then the deadline has transpired.
124    *
125    * @deprecated This method is deprecated because it encourages code that is
126    * susceptible to overflow bugs.
127    */
128   @SuppressWarnings("WeakerAccess")
129   @Deprecated(forRemoval = true)
130   public long getTimeLeft(long deadline) {
131     return deadline - NanoClock.nanoTime();
132   }
133 
134   /**
135    * Get the unit of precision for the underlying clock, that is used by
136    * {@link #getDeadline()} and {@link #getTimeLeft(long)}.
137    * @return TimeUnit The unit of precision used by the clock in this Timeout.
138    */
139   public TimeUnit getBaseUnit() {
140     return TimeUnit.NANOSECONDS;
141   }
142 
143   @Override
144   public int hashCode() {
145     return 31 * (1 + (int) (timeoutBase ^ (timeoutBase >>> 32)));
146   }
147 
148   /**
149    * Timeouts of equivalent duration are equal, even if they were constructed
150    * with different units.
151    * @return `true` if this Timeout value is equal to the given
152    * Timeout value, `false` otherwise.
153    */
154   @Override
155   public boolean equals(Object obj) {
