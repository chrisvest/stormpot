16  package stormpot;
17  
18  /**
19   * A reference to a pooled object.
20   */
21  public class Pooled<T> extends BasePoolable implements Poolable, AutoCloseable {
22    /**
23     * The object managed by this pooled instance.
24     */
25    public final T object;
26  
27    /**
28     * Create a pooled object for the given slot and object to be pooled.
29     *
30     * @param slot The slot an object is being allocated for.
31     * @param object The object this pooled instance represents.
32     */
33    public Pooled(Slot slot, T object) {
34      super(slot);
35      this.object = object;
36    }
37  
38    /**
39     * `Pooled` implements {@link AutoCloseable} as a convenient way to release
40     * claimed objects back to the pool, using the try-with-resources syntax.
41     */
42    @Override
43    public void close() {
44      release();
45    }
46  
47    @Override
48    public String toString() {
49      return "Pooled[" + object + "]";
50    }
51  }
