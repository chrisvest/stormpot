16  package stormpot;
17  
18  import java.util.concurrent.TimeUnit;
19  
20  final class TimingReallocatorAdaptor<T extends Poolable>
21      extends TimingReallocatingAdaptor<T>
22      implements Reallocator<T> {
23    TimingReallocatorAdaptor(
24        Reallocator<T> allocator, MetricsRecorder metricsRecorder) {
25      super(allocator, metricsRecorder);
26    }
27  
28    @Override
29    public T reallocate(Slot slot, T poolable) throws Exception {
30      long start = System.nanoTime();
31      try {
32        T obj = ((Reallocator<T>) allocator).reallocate(slot, poolable);
33        long elapsedNanos = System.nanoTime() - start;
34        long milliseconds = TimeUnit.NANOSECONDS.toMillis(elapsedNanos);
35        metricsRecorder.recordReallocationLatencySampleMillis(milliseconds);
36        return obj;
37      } catch (Exception e) {
38        long elapsedNanos = System.nanoTime() - start;
39        long milliseconds = TimeUnit.NANOSECONDS.toMillis(elapsedNanos);
40        metricsRecorder.recordReallocationFailureLatencySampleMillis(milliseconds);
41        throw e;
42      }
43    }
44  }
