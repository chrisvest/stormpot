16  package stormpot;
17  
18  /**
19   * Objects contained in a {@link Pool} must implement the Poolable interface
20   * and adhere to its contract.
21   * <p>
22   * Pools call {@link Allocator#allocate(Slot) allocate} on Allocators with the
23   * specific {@link Slot} that they want a Poolable allocated for. The Slot
24   * represents a location in the pool, that can fit an object and make it
25   * available for others to claim.
26   * <p>
27   * The contract of the Poolable interface is, that when {@link #release()} is
28   * called on the Poolable, it must in turn call {@link Slot#release(Poolable)}
29   * on the specific Slot object that it was allocated with, giving itself as the
30   * Poolable parameter.
31   * <p>
32   * The simplest possible correct implementation of the Poolable interface looks
33   * like this:
34   * <pre><code> public class GenericPoolable implements Poolable {
35   *   private final Slot slot;
36   *   public GenericPoolable(Slot slot) {
37   *     this.slot = slot;
38   *   }
39   *   
40   *   public void release() {
41   *     slot.release(this);
42   *   }
43   * }</code></pre>
44   * <p>
45   * See also <a href="package-summary.html#memory-effects-and-threading">
46   * Memory Effects and Threading</a> for details of the behaviour in concurrent
47   * programs.
48   * 
49   * @author Chris Vest &lt;mr.chrisvest@gmail.com&gt;
50   *
51   */
52  public interface Poolable {
53    /**
54     * Release this Poolable object back into the pool from where it came,
55     * so that others can claim it, or the pool deallocate it if it has
56     * expired.
57     * <p>
58     * A call to this method MUST delegate to a call to
59     * {@link Slot#release(Poolable)} on the Slot object for which this
60     * Poolable was allocated, giving itself as the Poolable parameter.
61     * <p>
62     * A Poolable can be released by a thread other than the one that claimed it.
63     * It can not, however, be released more than once per claim. The feature of
64     * permitting releases from threads other than the claiming thread might be
65     * useful in message-passing architectures and the like, but it is not
66     * generally recommendable because it makes it more complicated to keep track
67     * of object lifecycles.
68     * <p>
69     * Great care must be taken, to ensure that Poolables are not used after they
70     * are released!
71     */
72    void release();
73  }
